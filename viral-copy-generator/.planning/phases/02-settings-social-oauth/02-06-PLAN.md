```typescript
import { useEffect, useState } from 'react'
import type { Session } from '@supabase/supabase-js'
import { supabase } from './lib/supabase'
import LoginPage from './pages/LoginPage'
import GeneratorPage from './pages/GeneratorPage'
import SettingsPage from './pages/SettingsPage'
import type { Screen } from './lib/types'

export default function App() {
  const [session, setSession] = useState&lt;Session | null&gt;(null)
  const [loading, setLoading] = useState(true)
  const [currentScreen, setCurrentScreen] = useState&lt;Screen&gt;('generator')
  // Banner state set from OAuth redirect params; cleared on next manual nav
  const [oauthBanner, setOauthBanner] = useState&lt;
    | { kind: 'connected'; platform: 'youtube' | 'instagram' | 'facebook'; warning?: string }
    | { kind: 'error'; reason: string }
    | null
  &gt;(null)

  useEffect(() =&gt; {
    supabase.auth.getSession().then(({ data: { session }, error }) =&gt; {
      if (error) console.warn('[auth] getSession error:', error.message)
      setSession(session)
      setLoading(false)
    })

    const { data: { subscription } } = supabase.auth.onAuthStateChange((_event, session) =&gt; {
      setSession(session)
    })
    return () =&gt; subscription.unsubscribe()
  }, [])

  // OAuth-redirect param handler — must run after auth render so SettingsPage can refetch.
  // Reads ?screen=settings&connected=...|error=...|warning=... and strips the URL.
  useEffect(() =&gt; {
    if (!session) return
    const params = new URLSearchParams(window.location.search)
    if (params.size === 0) return

    const screen = params.get('screen')
    const connected = params.get('connected')
    const error = params.get('error')
    const warning = params.get('warning')

    if (screen === 'settings') setCurrentScreen('settings')
    if (connected === 'youtube' || connected === 'instagram' || connected === 'facebook') {
      setOauthBanner({ kind: 'connected', platform: connected, warning: warning ?? undefined })
    } else if (error) {
      setOauthBanner({ kind: 'error', reason: error })
    }

    // Strip params so a refresh does not re-trigger
    window.history.replaceState({}, '', window.location.pathname)
  }, [session])

  if (loading) return null
  if (!session) return &lt;LoginPage /&gt;

  if (currentScreen === 'settings') {
    return (
      &lt;SettingsPage
        onNavigate={setCurrentScreen}
        oauthBanner={oauthBanner}
        clearBanner={() =&gt; setOauthBanner(null)}
      /&gt;
    )
  }
  return &lt;GeneratorPage onNavigate={setCurrentScreen} /&gt;
}
```

```typescript
import { useCallback, useEffect, useState } from 'react'
import { supabase } from '../lib/supabase'
import { apiFetch } from '../lib/api'
import {
  AI_PROVIDERS, ALL_PLATFORMS, NICHES,
  type Screen, type SettingsResponse, type AIProvider, type Platform,
} from '../lib/types'

interface Props {
  onNavigate: (s: Screen) =&gt; void
  oauthBanner:
    | { kind: 'connected'; platform: 'youtube' | 'instagram' | 'facebook'; warning?: string }
    | { kind: 'error'; reason: string }
    | null
  clearBanner: () =&gt; void
}

const PLATFORM_LABELS: Record&lt;Platform, string&gt; = {
  youtube: 'YouTube',
  instagram: 'Instagram',
  tiktok: 'TikTok',
  facebook: 'Facebook',
  x: 'X / Twitter',
}

export default function SettingsPage({ onNavigate, oauthBanner, clearBanner }: Props) {
  const [data, setData] = useState&lt;SettingsResponse | null&gt;(null)
  const [loading, setLoading] = useState(true)
  const [saving, setSaving] = useState(false)
  const [error, setError] = useState&lt;string | null&gt;(null)
  const [keyDraft, setKeyDraft] = useState('')

  const refetch = useCallback(async () =&gt; {
    setLoading(true)
    try {
      const res = await apiFetch('/settings')
      if (!res.ok) throw new Error(`GET /settings ${res.status}`)
      setData(await res.json())
      setError(null)
    } catch (e) {
      setError((e as Error).message)
    } finally {
      setLoading(false)
    }
  }, [])

  useEffect(() =&gt; { void refetch() }, [refetch])

  async function patch(body: Record&lt;string, unknown&gt;): Promise&lt;void&gt; {
    setSaving(true)
    try {
      const res = await apiFetch('/settings', {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body),
      })
      if (!res.ok) {
        const j = await res.json().catch(() =&gt; ({} as { error?: string }))
        throw new Error(j.error ?? `PATCH /settings ${res.status}`)
      }
      await refetch()
    } catch (e) {
      setError((e as Error).message)
    } finally {
      setSaving(false)
    }
  }

  function connectPlatform(provider: 'youtube' | 'instagram' | 'facebook'): void {
    // COOP-safe redirect via JSON contract:
    // The backend's /api/auth/{provider}/connect returns 200 JSON { auth_url } (NOT 302).
    // We CANNOT use 302 + Location-read for an authenticated XHR — CORS hides the
    // Location header on cross-origin opaqueredirect responses (`response.headers.get('location')`
    // returns null in real browsers). Reading JSON and calling window.location.assign(auth_url)
    // is the only contract that works without a popup (popups are killed by COOP same-origin).
    initiateConnect(provider).catch((e) =&gt; setError((e as Error).message))
  }

  async function initiateConnect(provider: 'youtube' | 'instagram' | 'facebook'): Promise&lt;void&gt; {
    const path = provider === 'youtube' ? '/auth/google/connect' : `/auth/${provider}/connect`
    const res = await apiFetch(path)
    if (!res.ok) throw new Error(`Could not start OAuth (${res.status}) — please retry`)
    const json = await res.json() as { auth_url?: string }
    if (!json.auth_url || typeof json.auth_url !== 'string') {
      throw new Error('Could not start OAuth — invalid response')
    }
    window.location.assign(json.auth_url)
  }

  async function disconnect(provider: 'youtube' | 'instagram' | 'facebook'): Promise&lt;void&gt; {
    setSaving(true)
    try {
      const res = await apiFetch(`/settings/connections/${provider}`, { method: 'DELETE' })
      if (!res.ok) throw new Error(`disconnect failed ${res.status}`)
      await refetch()
    } catch (e) {
      setError((e as Error).message)
    } finally {
      setSaving(false)
    }
  }

  function togglePlatform(p: Platform): void {
    if (!data) return
    const enabled = data.enabled_platforms.includes(p)
      ? data.enabled_platforms.filter((x) =&gt; x !== p)
      : [...data.enabled_platforms, p]
    void patch({ enabled_platforms: enabled })
  }

  return (
    &lt;div className="flex h-[100dvh] flex-col bg-zinc-950 text-white"&gt;
      &lt;header className="flex items-center justify-between px-4 py-3"&gt;
        &lt;span className="font-bold"&gt;Settings&lt;/span&gt;
        &lt;div className="flex items-center gap-2"&gt;
          &lt;button
            onClick={() =&gt; onNavigate('generator')}
            className="rounded bg-zinc-800 px-3 py-1.5 text-sm hover:bg-zinc-700"
          &gt;
            Generator
          &lt;/button&gt;
          &lt;button
            onClick={() =&gt; supabase.auth.signOut()}
            className="rounded bg-zinc-800 px-3 py-1.5 text-sm hover:bg-zinc-700"
          &gt;
            Sign out
          &lt;/button&gt;
        &lt;/div&gt;
      &lt;/header&gt;

      &lt;main className="flex-1 overflow-y-auto px-4 pb-[env(safe-area-inset-bottom)]"&gt;
        {oauthBanner &amp;&amp; (
          &lt;div className={`mb-4 rounded p-3 text-sm ${
            oauthBanner.kind === 'connected'
              ? oauthBanner.warning ? 'bg-yellow-900/40 text-yellow-200' : 'bg-emerald-900/40 text-emerald-200'
              : 'bg-red-900/40 text-red-200'
          }`}&gt;
            {oauthBanner.kind === 'connected'
              ? oauthBanner.warning === 'no_facebook_page'
                ? 'Facebook connected, but no Page found. Create a Facebook Page to enable Reels uploads.'
                : `${PLATFORM_LABELS[oauthBanner.platform]} connected.`
              : oauthBanner.reason === 'instagram_personal_account'
              ? 'Personal Instagram accounts cannot publish. Switch to Business or Creator and retry.'
              : `Connection failed (${oauthBanner.reason}). Please try again.`}
            &lt;button onClick={clearBanner} className="ml-3 underline"&gt;dismiss&lt;/button&gt;
          &lt;/div&gt;
        )}

        {error &amp;&amp; &lt;div className="mb-4 rounded bg-red-900/40 p-3 text-sm text-red-200"&gt;{error}&lt;/div&gt;}

        {loading || !data ? (
          &lt;div className="py-8 text-center text-zinc-400"&gt;Loading…&lt;/div&gt;
        ) : (
          &lt;div className="space-y-6 py-4"&gt;
            {/* AI Provider + API key (SETTINGS-01) */}
            &lt;section&gt;
              &lt;h2 className="mb-2 font-bold"&gt;AI Provider&lt;/h2&gt;
              &lt;select
                value={data.ai_provider}
                onChange={(e) =&gt; void patch({ ai_provider: e.target.value as AIProvider })}
                disabled={saving}
                className="mb-2 rounded bg-zinc-800 px-3 py-2 text-sm"
              &gt;
                {AI_PROVIDERS.map((p) =&gt; &lt;option key={p} value={p}&gt;{p}&lt;/option&gt;)}
              &lt;/select&gt;
              &lt;div className="text-sm text-zinc-400"&gt;
                Current key: {data.api_key_masked ?? &lt;em&gt;not set&lt;/em&gt;}
              &lt;/div&gt;
              &lt;form
                onSubmit={(e) =&gt; { e.preventDefault(); if (keyDraft.length &gt; 0) { void patch({ api_key: keyDraft }); setKeyDraft('') } }}
                className="mt-2 flex gap-2"
              &gt;
                &lt;input
                  type="password"
                  value={keyDraft}
                  onChange={(e) =&gt; setKeyDraft(e.target.value)}
                  placeholder="Paste new API key"
                  className="flex-1 rounded bg-zinc-800 px-3 py-2 text-sm"
                /&gt;
                &lt;button type="submit" disabled={saving || keyDraft.length === 0}
                        className="rounded bg-emerald-700 px-3 py-2 text-sm font-bold hover:bg-emerald-600 disabled:opacity-40"&gt;
                  Save
                &lt;/button&gt;
              &lt;/form&gt;
            &lt;/section&gt;

            {/* Default niche (SETTINGS-02) */}
            &lt;section&gt;
              &lt;h2 className="mb-2 font-bold"&gt;Default Niche&lt;/h2&gt;
              &lt;select
                value={data.default_niche}
                onChange={(e) =&gt; void patch({ default_niche: e.target.value })}
                disabled={saving}
                className="rounded bg-zinc-800 px-3 py-2 text-sm"
              &gt;
                {NICHES.map((n) =&gt; &lt;option key={n} value={n}&gt;{n}&lt;/option&gt;)}
              &lt;/select&gt;
            &lt;/section&gt;

            {/* Platform toggles (SETTINGS-03) */}
            &lt;section&gt;
              &lt;h2 className="mb-2 font-bold"&gt;Enabled Platforms&lt;/h2&gt;

                      disabled={saving}
                    /&gt;
                    {PLATFORM_LABELS[p]}
                  &lt;/label&gt;
                ))}
              &lt;/div&gt;
            &lt;/section&gt;

            {/* OAuth connections (SETTINGS-04, 05, 06, 09) */}
            &lt;section&gt;
              &lt;h2 className="mb-2 font-bold"&gt;Connections&lt;/h2&gt;
              &lt;div className="space-y-2"&gt;
```
