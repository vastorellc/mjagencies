```typescript
// backend/src/middleware/admin.ts
// ADMIN-01: Admin guard — enforces role: 'admin' JWT claim on all /api/admin/* routes.
// This middleware is ALWAYS applied AFTER authMiddleware — res.locals.user is guaranteed
// to be set when this runs. It is a SEPARATE file from auth.ts (CLAUDE.md: never mix concerns).
import type { Request, Response, NextFunction } from 'express'
import type { User } from '@supabase/supabase-js'

// Extend Express locals so TypeScript knows res.locals.user is a Supabase User.
// This matches the shape set by authMiddleware in auth.ts.
declare global {
  namespace Express {
    interface Locals {
      user: User
      userId: string
    }
  }
}

export function adminMiddleware(
  _req: Request,
  res: Response,
  next: NextFunction,
): void {
  // res.locals.user is guaranteed by authMiddleware which runs first at app.use('/api', authMiddleware).
  // app_metadata.role is set server-side via Supabase service role key — cannot be forged by JWT claim body.
  const role = res.locals.user?.app_metadata?.role as string | undefined
  if (role !== 'admin') {
    // ADMIN-10 baseline: always 403, never expose why (enumeration risk)
    res.status(403).json({ error: 'Forbidden' })
    return
  }
  next()
}
```

Important: adminMiddleware is SYNCHRONOUS (no async/await) — it only reads res.locals which is already populated. This avoids the Express 5 async error forwarding edge case for sync handlers.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client JWT → authMiddleware | JWT verified by Supabase; app_metadata injected server-side — cannot be forged in token body |
| authMiddleware → adminMiddleware | res.locals.user already validated; admin check reads app_metadata.role |
| /api/admin/* → admin routes | Double gate: authMiddleware (401 without JWT) then adminMiddleware (403 without role) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-01 | Elevation of Privilege | adminMiddleware | mitigate | Read role from `res.locals.user.app_metadata.role` (set server-side by Supabase service key only); never accept role from request body, query, or headers |
| T-08-02 | Spoofing | app_metadata.role | mitigate | Supabase JWTs are signed with project JWT secret; app_metadata is a Supabase-controlled field — not in the JWT claims body editable by clients |
| T-08-03 | Elevation of Privilege | adminRouter mount order | mitigate | adminRouter mounted AFTER `app.use('/api', authMiddleware)` — every admin request passes authMiddleware first; adminMiddleware is a second gate inside the router |
| T-08-04 | Information Disclosure | 403 response body | accept | Returns `{ error: 'Forbidden' }` only — no disclosure of why (role mismatch vs missing vs wrong value) |
</threat_model>
