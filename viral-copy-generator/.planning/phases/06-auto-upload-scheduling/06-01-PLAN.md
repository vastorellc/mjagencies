/**
 * STORE-03: Delete a single file after successful social upload.
 * Path-traversal guard: resolved path must be inside UPLOADS_ROOT.
 */
export async function deleteFile(filePath: string): Promise<void> {
  const root = path.resolve(UPLOADS_ROOT)
  const resolved = path.resolve(filePath)
  if (!resolved.startsWith(root + path.sep)) {
    throw new Error(`deleteFile: path traversal rejected: ${filePath}`)
  }
  await unlink(resolved)
  console.log(`[storage] deleted: ${resolved}`)
}

```typescript
import { Router, type Request, type Response } from 'express'
import multer from 'multer'
import { randomUUID } from 'node:crypto'
import path from 'node:path'
import { mkdir } from 'node:fs/promises'
import { UPLOADS_ROOT } from '../lib/storage.js'
import { getBoss } from '../lib/boss.js'
import { db } from '../db/index.js'
import { platform_posts } from '../db/schema.js'
import { eq, and } from 'drizzle-orm'

export const uploadRouter = Router()

const MAX_FILE_SIZE = 260 * 1024 * 1024 // 260 MB (250 MB limit + 10 MB headroom for multipart)

// multer: store to /var/uploads/{userId}/{uuid}.mp4
// diskStorage because memStorage would OOM on large files
const storage = multer.diskStorage({
  destination: async (_req, _file, cb) => {
    // userId injected by authMiddleware into res.locals — but multer destination
    // runs before route handler, so we must use a temp dir and rename later.
    // Strategy: write to UPLOADS_ROOT/tmp/{uuid} then move in route handler.
    const tmpDir = path.resolve(UPLOADS_ROOT, 'tmp')
    await mkdir(tmpDir, { recursive: true })
    cb(null, tmpDir)
  },
  filename: (_req, _file, cb) => {
    cb(null, `${randomUUID()}.mp4`)
  },
})

const upload = multer({
  storage,
  limits: { fileSize: MAX_FILE_SIZE },
  fileFilter: (_req, file, cb) => {
    if (!file.mimetype.startsWith('video/')) {
      cb(new Error('only_video_files'))
    } else {
      cb(null, true)
    }
  },
})

/**
 * POST /api/upload/file
 * Body: multipart/form-data, field name "video"
 * Returns: { fileId: string, publicUrl: string }
 * Writes to /var/uploads/{userId}/{uuid}.mp4
 * Requires: VPS_PUBLIC_URL env var
 */
uploadRouter.post('/file', upload.single('video'), async (req: Request, res: Response) => {
  if (!req.file) {
    res.status(400).json({ error: 'no_file' })
    return
  }

  const userId = res.locals.userId as string
  const vpsUrl = process.env.VPS_PUBLIC_URL
  if (!vpsUrl) {
    res.status(500).json({ error: 'VPS_PUBLIC_URL not configured' })
    return
  }

  // Move from tmp dir to user-namespaced directory
  const { rename } = await import('node:fs/promises')
  const fileId = path.basename(req.file.filename, '.mp4') // uuid portion
  const userDir = path.resolve(UPLOADS_ROOT, userId)
  await mkdir(userDir, { recursive: true })
  const destPath = path.resolve(userDir, `${fileId}.mp4`)

  // Path traversal guard
  const root = path.resolve(UPLOADS_ROOT)
  if (!destPath.startsWith(root + path.sep)) {
    res.status(400).json({ error: 'invalid_path' })
    return
  }

  await rename(req.file.path, destPath)

  const publicUrl = `${vpsUrl}/uploads/${userId}/${fileId}.mp4`
  res.json({ fileId, publicUrl })
})

/**
 * POST /api/upload/schedule
 * Body JSON: { postId, platform, scheduledAt? }
 * scheduledAt: ISO-8601 UTC string; omit for immediate dispatch.
 * Enqueues a pg-boss job with the upload payload.
 * Instagram 100 MB gate enforced here (STORE-05, AUTOUP-02).
 */
interface ScheduleBody {
  postId: string
  platform: 'youtube' | 'instagram' | 'facebook' | 'tiktok'
  fileId: string
  filePath: string
  publicUrl: string
  caption: string
  hashtags: string[]
  scheduledAt?: string
  fileSizeBytes?: number
}

uploadRouter.post('/schedule', async (req: Request, res: Response) => {
  const userId = res.locals.userId as string
  const body = req.body as ScheduleBody

  // Validate required fields
  if (!body.postId || !body.platform || !body.fileId || !body.filePath || !body.publicUrl) {
    res.status(400).json({ error: 'missing_required_fields' })
    return
  }

  const VALID_PLATFORMS = ['youtube', 'instagram', 'facebook', 'tiktok']
  if (!VALID_PLATFORMS.includes(body.platform)) {
    res.status(400).json({ error: 'invalid_platform' })
    return
  }

  // STORE-05: Instagram 100 MB gate — reject before queuing
  if (body.platform === 'instagram' && body.fileSizeBytes !== undefined) {
    if (body.fileSizeBytes > 100 * 1024 * 1024) {
      res.status(413).json({ error: 'instagram_file_too_large', message: 'Instagram: max 100 MB. Please compress the video before uploading.' })
      return
    }
  }

  // Find the platform_posts row to get its id for status updates
  const platformPostRows = await db
    .select({ id: platform_posts.id })
    .from(platform_posts)
    .where(
      and(
        eq(platform_posts.post_id, body.postId),
        eq(platform_posts.user_id, userId),
        eq(platform_posts.platform, body.platform),
      )
    )
    .limit(1)

  if (platformPostRows.length === 0) {
    res.status(404).json({ error: 'platform_post_not_found' })
    return
  }

  const platformPostId = platformPostRows[0].id

  const boss = await getBoss()
  const jobName = `upload-${body.platform}`

  const payload = {
    userId,
    postId: body.postId,
    platformPostId,
    fileId: body.fileId,
    filePath: body.filePath,
    publicUrl: body.publicUrl,
    platform: body.platform,
    caption: body.caption,
    hashtags: body.hashtags,
    scheduledAt: body.scheduledAt ?? new Date().toISOString(),
  }

  // Update platform_posts.upload_status to 'uploading'
  await db
    .update(platform_posts)
    .set({ upload_status: 'uploading' })
    .where(eq(platform_posts.id, platformPostId))

  const sendOptions = body.scheduledAt
    ? { startAfter: new Date(body.scheduledAt).toISOString() }
    : {}

  await boss.send(jobName, payload, sendOptions)

  res.json({ ok: true, platformPostId })
})
```

```typescript
// backend/src/lib/upload-worker.ts
// pg-boss worker registration for auto-upload jobs (Phase 6)
// Worker implementations: see upload-youtube.ts, upload-instagram.ts, upload-facebook.ts
import { PgBoss } from 'pg-boss'
import { db } from '../db/index.js'
import { platform_posts } from '../db/schema.js'
import { eq } from 'drizzle-orm'

export interface UploadJobPayload {
  userId: string
  postId: string
  platformPostId: string
  fileId: string
  filePath: string
  publicUrl: string
  platform: 'youtube' | 'instagram' | 'facebook' | 'tiktok'
  caption: string
  hashtags: string[]
  scheduledAt: string
}

const UPLOAD_QUEUES = ['upload-youtube', 'upload-instagram', 'upload-facebook', 'upload-tiktok'] as const

/**
 * Update platform_posts.upload_status + optional error_message.
 * Called by each worker on success / failure.
 * Supabase Realtime pushes the change to the frontend (AUTOUP-08).
 */
export async function updateUploadStatus(
  platformPostId: string,
  status: 'uploading' | 'posted' | 'failed',
  errorMessage?: string,
): Promise<void> {
  await db
    .update(platform_posts)
    .set({
      upload_status: status,
      ...(status === 'posted' ? { posted_at: new Date() } : {}),
      ...(errorMessage ? { error_message: errorMessage } : {}),
    })
    .where(eq(platform_posts.id, platformPostId))
}

export async function registerUploadWorkers(boss: PgBoss): Promise<void> {
  // pg-boss v12: createQueue() before work() — FK constraint on queue name
  for (const queueName of UPLOAD_QUEUES) {
    await boss.createQueue(queueName)
  }

  // YouTube worker — implementation filled in Plan 06-02
  await boss.work<UploadJobPayload>('upload-youtube', async (job) => {
    const { data } = job
    const { uploadYouTube } = await import('./upload-youtube.js')
    await uploadYouTube(data)
  })

  // Instagram worker — implementation filled in Plan 06-02
  await boss.work<UploadJobPayload>('upload-instagram', async (job) => {
    const { data } = job
    const { uploadInstagram } = await import('./upload-instagram.js')
    await uploadInstagram(data)
  })

  // Facebook worker — implementation filled in Plan 06-02
  await boss.work<UploadJobPayload>('upload-facebook', async (job) => {
    const { data } = job
    const { uploadFacebook } = await import('./upload-facebook.js')
    await uploadFacebook(data)
  })

  // TikTok worker — hidden behind isTikTokApproved flag (AUTOUP-04)
  await boss.work<UploadJobPayload>('upload-tiktok', async (job) => {
    const { data } = job
    const isTikTokApproved = process.env.TIKTOK_APPROVED === 'true'
    if (!isTikTokApproved) {
      // Fail job gracefully — not an error, just not yet approved
      await updateUploadStatus(data.platformPostId, 'failed', 'TikTok API approval pending')
      return
    }
    const { uploadTikTok } = await import('./upload-tiktok.js')
    await uploadTikTok(data)
  })

  console.log('[pg-boss] upload workers registered: youtube, instagram, facebook, tiktok')
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → POST /api/upload/file | Authenticated user sends video file — multer must enforce size and type |
| backend → VPS filesystem | File write paths must be validated to prevent path traversal |
| job payload → worker | Worker receives userId/platformPostId from DB-resolved values, not raw client input |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01 | Tampering | POST /api/upload/file — file path write | mitigate | `path.resolve(destPath).startsWith(UPLOADS_ROOT + sep)` guard before rename |
| T-06-02 | Tampering | deleteFile() — path traversal | mitigate | Same `path.resolve().startsWith(root + sep)` check; throw on violation |
| T-06-03 | Elevation of Privilege | POST /api/upload/schedule — userId from body | mitigate | `userId` always from `res.locals.userId` (authMiddleware); never from `req.body` |
| T-06-04 | Spoofing | pg-boss job payload | accept | Payload assembled server-side from verified userId; job worker trusts payload userId for DB writes |
| T-06-05 | DoS | Large file upload | mitigate | multer `limits.fileSize: 260 MB` hard-stops before disk write; multer returns 413 |
