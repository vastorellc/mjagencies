Brightness algorithm (RESEARCH §"Pattern 8"):
- Draw frame onto a 64×64 canvas via drawImage
- Sample every 4th pixel (so 256 samples / frame)
- BT.601 luma = 0.299r + 0.587g + 0.114b
- Per-frame brightness = sum / count / 255 (0..1 range)
- Aggregate brightnessScore = mean across frames; 0 if no frames

    // ============================================================================
    // INTERNAL: audio analysis — Pattern 7 (OfflineAudioContext + Meyda)
    // ============================================================================

    // Calibration knobs — tuned against fixture videos in 03-08
    const AUDIO_BUFFER_SIZE = 1024
    const BEAT_FLUX_THRESHOLD = 0.05
    const SILENCE_RMS_THRESHOLD = 0.02
    const MIN_SILENCE_GAP_SEC = 1.5

    interface AudioSignals {
      audioEnergy: number
      beatPresent: boolean
      silenceGapsSec: number[]
      hasAudio: boolean
    }

    const EMPTY_AUDIO: AudioSignals = {
      audioEnergy: 0,
      beatPresent: false,
      silenceGapsSec: [],
      hasAudio: false,
    }

    async function analyseAudio(file: File): Promise<AudioSignals> {
      const arrayBuf = await file.arrayBuffer()

      // OfflineAudioContext is the right tool for static-file analysis (NOT AnalyserNode).
      // We construct a tiny temporary context just to call decodeAudioData; sampleRate 44100 is conventional.
      const tmpCtx = new OfflineAudioContext(1, 1, 44100)
      let audioBuf: AudioBuffer
      try {
        // .slice(0) — ArrayBuffer is consumed by decodeAudioData in some implementations
        audioBuf = await tmpCtx.decodeAudioData(arrayBuf.slice(0))
      } catch {
        // Pitfall 6 — file has no audio track or unsupported codec; return defaults, never throw.
        return EMPTY_AUDIO
      }
      if (audioBuf.numberOfChannels === 0) return EMPTY_AUDIO

      const channelData = audioBuf.getChannelData(0)
      const sampleRate = audioBuf.sampleRate
      const energies: number[] = []
      const fluxes: number[] = []
      const rmsSeries: number[] = []

      for (let i = 0; i + AUDIO_BUFFER_SIZE < channelData.length; i += AUDIO_BUFFER_SIZE) {
        const window = channelData.slice(i, i + AUDIO_BUFFER_SIZE)
        const features = Meyda.extract(['energy', 'rms', 'spectralFlux'], window) as
          | { energy: number; rms: number; spectralFlux: number }
          | null
        if (!features) continue
        if (Number.isFinite(features.energy)) energies.push(features.energy)
        if (Number.isFinite(features.rms)) rmsSeries.push(features.rms)
        if (Number.isFinite(features.spectralFlux)) fluxes.push(features.spectralFlux)
      }

      const meanEnergy = energies.length === 0
        ? 0
        : energies.reduce((a, b) => a + b, 0) / energies.length
      // Meyda 'energy' range is roughly 0..bufferSize (sum of squares); normalise to 0..1
      const audioEnergy = Math.min(1, meanEnergy / AUDIO_BUFFER_SIZE)

      const meanFlux = fluxes.length === 0
        ? 0
        : fluxes.reduce((a, b) => a + b, 0) / fluxes.length
      const beatPresent = meanFlux > BEAT_FLUX_THRESHOLD

      // Silence gaps — contiguous low-RMS windows ≥ MIN_SILENCE_GAP_SEC
      const windowSec = AUDIO_BUFFER_SIZE / sampleRate
      const gaps: number[] = []
      let runStart: number | null = null
      for (let i = 0; i < rmsSeries.length; i++) {
        if (rmsSeries[i] < SILENCE_RMS_THRESHOLD) {
          if (runStart === null) runStart = i
        } else if (runStart !== null) {
          const dur = (i - runStart) * windowSec
          if (dur >= MIN_SILENCE_GAP_SEC) gaps.push(dur)
          runStart = null
        }
      }
      if (runStart !== null) {
        const dur = (rmsSeries.length - runStart) * windowSec
        if (dur >= MIN_SILENCE_GAP_SEC) gaps.push(dur)
      }

      return { audioEnergy, beatPresent, silenceGapsSec: gaps, hasAudio: true }
    }

    // ============================================================================
    // INTERNAL: brightness — Pattern 8 (BT.601 luma)
    // ============================================================================

    function computeBrightnessForCanvas(source: HTMLCanvasElement): number {
      const small = document.createElement('canvas')
      small.width = 64
      small.height = 64
      const ctx = small.getContext('2d')
      if (!ctx) return 0
      ctx.drawImage(source, 0, 0, 64, 64)
      const data = ctx.getImageData(0, 0, 64, 64).data

      let sum = 0
      let count = 0
      for (let i = 0; i < data.length; i += 4 * 4) {
        const r = data[i]
        const g = data[i + 1]
        const b = data[i + 2]
        sum += 0.299 * r + 0.587 * g + 0.114 * b
        count++
      }
      if (count === 0) return 0
      return sum / count / 255
    }

    function computeBrightnessAcrossFrames(frames: HTMLCanvasElement[]): number {
      if (frames.length === 0) return 0
      let total = 0
      for (const c of frames) total += computeBrightnessForCanvas(c)
      return Math.min(1, Math.max(0, total / frames.length))
    }

    import { describe, it, expect, afterEach, beforeEach, vi } from 'vitest'
    import { analyse, __resetEngineForTests } from './engine'
    import { loadFixture } from '../../test/setup'

    describe('engine — ANALYSIS-07 no backend roundtrip during analyse()', () => {
      let originalFetch: typeof fetch
      const calls: string[] = []

      beforeEach(() => {
        originalFetch = window.fetch
        calls.length = 0
        window.fetch = vi.fn(async (...args: Parameters<typeof fetch>) => {
          const url = typeof args[0] === 'string'
            ? args[0]
            : (args[0] as Request | URL).toString()
          calls.push(url)
          return originalFetch.apply(window, args)
        }) as unknown as typeof fetch
      })

      afterEach(() => {
        window.fetch = originalFetch
        __resetEngineForTests()
      })

      it('does not fetch from backend / Supabase / localhost during analyse', async () => {
        const file = await loadFixture('with-face.mp4').catch(() => null)
        if (!file) return
        await analyse(file)

        // Whitelist: ffmpeg-core CDN, MediaPipe CDN, blob: URIs (toBlobURL artifacts), data: URIs (frames)
        const allowedHosts = [
          'cdn.jsdelivr.net',  // covers @ffmpeg/core + @mediapipe/face_detection
          'storage.googleapis.com',  // TF.js model storage (coco-ssd, mediapipe internals)
          'tfhub.dev',
          'unpkg.com',  // alternative CDN that some TF.js models default to
        ]

        const violations: string[] = []
        for (const url of calls) {
          if (url.startsWith('blob:') || url.startsWith('data:')) continue
          let parsed: URL
          try {
            parsed = new URL(url)
          } catch {
            // Relative URL → almost certainly a backend hit
            violations.push(url)
            continue
          }
          if (!allowedHosts.some((h) => parsed.host.endsWith(h))) {
            violations.push(url)
          }
        }

        // Fail loudly if anything points at backend / supabase / our own dev server
        expect(violations, `unexpected non-whitelisted fetches: ${JSON.stringify(violations)}`)
          .toEqual([])
      }, 180_000)
    })
