    ```typescript
    export const HARD_LIMIT_BYTES = 250 * 1024 * 1024
    export const ADVISORY_BYTES = 200 * 1024 * 1024
    export const ACCEPTED_MIME: ReadonlyArray<string> = ['video/mp4', 'video/quicktime']
    export const ACCEPT_ATTR = 'video/mp4,video/quicktime'
    export const DESCRIPTION_MAX_CHARS = 280

    export type ValidateResult =
      | { ok: true; file: File }
      | { ok: false; code: 'too_large' | 'bad_type' | 'empty'; message: string }

    export function validateUploadFile(file: File): ValidateResult {
      if (file.size === 0) {
        return { ok: false, code: 'empty', message: 'File appears to be empty.' }
      }
      if (file.size > HARD_LIMIT_BYTES) {
        return {
          ok: false,
          code: 'too_large',
          message: 'File over 250 MB — compress with HandBrake or CapCut and try again.',
        }
      }
      // file.type comes from the browser MIME sniff, NOT extension — defence in depth.
      if (!ACCEPTED_MIME.includes(file.type)) {
        return { ok: false, code: 'bad_type', message: 'Only MP4 and MOV files are supported.' }
      }
      return { ok: true, file }
    }

    export function formatBytes(bytes: number): string {
      if (bytes < 1024) return `${bytes} B`
      const kb = bytes / 1024
      if (kb < 1024) return `${Math.round(kb)} KB`
      const mb = kb / 1024
      if (mb < 100) return `${mb.toFixed(1)} MB`
      return `${Math.round(mb)} MB`
    }

    export function isTouchDevice(): boolean {
      if (typeof navigator === 'undefined') return false
      // Either pointer-coarse media query OR maxTouchPoints > 0 catches phones, tablets, hybrid laptops in tablet mode
      const coarse = typeof window !== 'undefined' && window.matchMedia?.('(pointer: coarse)').matches
      return Boolean(coarse) || navigator.maxTouchPoints > 0
    }

    export function isNarrowViewport(): boolean {
      if (typeof window === 'undefined') return false
      return window.innerWidth < 768
    }
    ```

    ```tsx
    import { useRef, useState, useEffect } from 'react'
    import {
      validateUploadFile,
      isTouchDevice,
      isNarrowViewport,
      ACCEPT_ATTR,
    } from '../lib/upload'

    interface Props {
      onFile: (file: File) => void
      onError: (message: string) => void
      className?: string
    }

    export default function UploadDropzone({ onFile, onError, className = '' }: Props) {
      const inputRef = useRef<HTMLInputElement>(null)
      const [isDragOver, setIsDragOver] = useState(false)
      const [touchOnly, setTouchOnly] = useState(false)

      useEffect(() => {
        // Compute on mount + on resize so a developer rotating their device updates UX
        const update = () => setTouchOnly(isTouchDevice() || isNarrowViewport())
        update()
        window.addEventListener('resize', update)
        return () => window.removeEventListener('resize', update)
      }, [])

      function handleFile(file: File | undefined | null) {
        if (!file) return
        const result = validateUploadFile(file)
        if (result.ok) onFile(result.file)
        else onError(result.message)
      }

      function onDrop(e: React.DragEvent<HTMLDivElement>) {
        e.preventDefault()
        setIsDragOver(false)
        const file = e.dataTransfer.files?.[0]
        handleFile(file)
      }

      function onDragOver(e: React.DragEvent<HTMLDivElement>) {
        e.preventDefault()
        setIsDragOver(true)
      }

      function onDragLeave() {
        setIsDragOver(false)
      }

      function onPick(e: React.ChangeEvent<HTMLInputElement>) {
        const file = e.target.files?.[0]
        handleFile(file)
        // Allow re-picking the same file (input would otherwise no-op)
        e.target.value = ''
      }

      function onKeyDown(e: React.KeyboardEvent<HTMLDivElement>) {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault()
          inputRef.current?.click()
        }
      }

      // Touch devices: only render a picker button — drag-drop is desktop-only per D-13.
      if (touchOnly) {
        return (
          <div className={`rounded-lg border border-zinc-800 bg-zinc-900 p-6 text-center ${className}`}>
            <p className="mb-3 text-sm text-zinc-300">Tap to pick a video</p>
            <button
              type="button"
              onClick={() => inputRef.current?.click()}
              className="rounded bg-zinc-100 px-4 py-2 text-sm font-bold text-zinc-900 hover:bg-zinc-200"
            >
              Choose video
            </button>
            <input
              ref={inputRef}
              type="file"
              accept={ACCEPT_ATTR}
              onChange={onPick}
              className="hidden"
              data-testid="upload-input"
            />
          </div>
        )
      }

      // Desktop: drag-drop card.
      return (
        <div
          role="button"
          tabIndex={0}
          aria-label="Upload video"
          onClick={() => inputRef.current?.click()}
          onKeyDown={onKeyDown}
          onDrop={onDrop}
          onDragOver={onDragOver}
          onDragLeave={onDragLeave}
          data-testid="dropzone"
          data-drag-over={isDragOver ? 'true' : 'false'}
          className={[
            'rounded-lg border-2 border-dashed p-8 text-center cursor-pointer transition-colors',
            'bg-zinc-900',
            isDragOver
              ? 'border-zinc-300 bg-zinc-800/60'
              : 'border-zinc-700 hover:border-zinc-500',
            className,
          ].join(' ')}
        >
          <p className="text-sm text-zinc-300">Drag a video here, or click to pick a file</p>
          <p className="mt-1 text-xs text-zinc-500">MP4 or MOV, up to 250 MB</p>
          <input
            ref={inputRef}
            type="file"
            accept={ACCEPT_ATTR}
            onChange={onPick}
            className="hidden"
            data-testid="upload-input"
          />
        </div>
      )
    }
    ```

    ```tsx
    import { useEffect, useRef, useState } from 'react'
    import { DESCRIPTION_MAX_CHARS, formatBytes } from '../lib/upload'

    export interface VideoMetadata {
      durationSec: number
      width: number
      height: number
      aspectRatio: number
    }

    interface Props {
      file: File
      description: string
      onDescriptionChange: (next: string) => void
      onMetadata?: (meta: VideoMetadata) => void
    }

    export default function VideoPreview({ file, description, onDescriptionChange, onMetadata }: Props) {
      const [thumbDataUrl, setThumbDataUrl] = useState<string | null>(null)
      const [meta, setMeta] = useState<VideoMetadata | null>(null)
      const videoRef = useRef<HTMLVideoElement>(null)
      const blobUrlRef = useRef<string | null>(null)

      useEffect(() => {
        // Reset state when file changes
        setMeta(null)
        setThumbDataUrl(null)

        const url = URL.createObjectURL(file)
        // Revoke any prior URL before storing new one (T-03-11)
        if (blobUrlRef.current) URL.revokeObjectURL(blobUrlRef.current)
        blobUrlRef.current = url

        const video = document.createElement('video')
        video.muted = true
        video.preload = 'metadata'
        video.playsInline = true
        video.src = url

        // Two-step per Pitfall 10
        video.onloadedmetadata = () => {
          const m: VideoMetadata = {
            durationSec: Number.isFinite(video.duration) ? video.duration : 0,
            width: video.videoWidth,
            height: video.videoHeight,
            aspectRatio: video.videoWidth > 0 ? video.videoWidth / video.videoHeight : 0,
          }
          setMeta(m)
          onMetadata?.(m)
          // Seek to 10% of duration to capture a meaningful thumbnail
          const seekTo = Math.max(0, m.durationSec * 0.1)
          video.currentTime = seekTo
        }

        video.onseeked = () => {
          const c = document.createElement('canvas')
          // Cap thumbnail width at 320px while preserving aspect ratio
          const w = Math.min(320, video.videoWidth || 320)
          const h = video.videoWidth > 0 ? Math.round((w / video.videoWidth) * video.videoHeight) : 180
          c.width = w
          c.height = h
          const ctx = c.getContext('2d')
          if (!ctx) return
          ctx.drawImage(video, 0, 0, w, h)
          try {
            setThumbDataUrl(c.toDataURL('image/jpeg', 0.8))
          } catch {
            // toDataURL may throw if the video is tainted (rare with same-origin blob URLs); silent — placeholder remains
          }
        }

        video.onerror = () => {
          // Leave thumbDataUrl null — VideoPreview shows placeholder, GeneratorPage may still allow Analyse
        }

        videoRef.current = video

        return () => {
          video.onloadedmetadata = null
          video.onseeked = null
          video.onerror = null
          if (blobUrlRef.current) {
            URL.revokeObjectURL(blobUrlRef.current)
            blobUrlRef.current = null
          }
        }
      }, [file, onMetadata])

      function metaStrip(): string {
        if (!meta) return `${formatBytes(file.size)}`
        const dur = meta.durationSec.toFixed(1)
        const ar = meta.aspectRatio > 0 ? meta.aspectRatio.toFixed(2) : '—'
        return `Duration: ${dur}s · ${meta.width}×${meta.height} · AR ${ar} · ${formatBytes(file.size)}`
      }

      return (
        <div className="space-y-4" data-testid="video-preview">
          <div className="rounded-lg border border-zinc-800 bg-zinc-900 overflow-hidden">
            <div className="aspect-video bg-zinc-950 flex items-center justify-center">
              {thumbDataUrl ? (
                <img src={thumbDataUrl} alt="Video thumbnail" className="max-h-full max-w-full" data-testid="thumbnail" />
              ) : (
                <span className="text-sm text-zinc-500" data-testid="thumbnail-placeholder">Loading preview…</span>
              )}
            </div>
            <div className="px-4 py-3 text-xs text-zinc-400" data-testid="metadata-strip">
              {metaStrip()}
            </div>
          </div>

          <label className="block">
            <span className="sr-only">Optional description</span>
            <textarea
              rows={2}
              maxLength={DESCRIPTION_MAX_CHARS}
              value={description}
              onChange={(e) => onDescriptionChange(e.target.value)}
              placeholder="Optional: brief description — helps AI when video is ambiguous"
              data-testid="description-textarea"
              className="w-full rounded-md border border-zinc-700 bg-zinc-900 px-3 py-2 text-sm text-zinc-100 placeholder:text-zinc-500 focus:border-zinc-500 focus:outline-none"
            />
          </label>
        </div>
      )
    }
    ```
