    ```typescript
    import { FFmpeg } from '@ffmpeg/ffmpeg'
    import { toBlobURL } from '@ffmpeg/util'
    import type { EngineSignals, AnalyseOptions, EnginePreflight, ProgressStep } from './types'

    // ============================================================================
    // PREFLIGHT — ANALYSIS-09 / D-11
    // ============================================================================

    /** Returns ok:true only when WebAssembly + SharedArrayBuffer + crossOriginIsolated all hold. */
    export function canRunEngine(): EnginePreflight {
      if (typeof WebAssembly === 'undefined') {
        return { ok: false, reason: 'WebAssembly not available in this browser' }
      }
      if (typeof SharedArrayBuffer === 'undefined') {
        return { ok: false, reason: 'SharedArrayBuffer required (cross-origin isolation missing)' }
      }
      if (typeof window !== 'undefined' && window.crossOriginIsolated !== true) {
        return { ok: false, reason: 'Cross-origin isolation not active — COOP/COEP misconfig' }
      }
      return { ok: true }
    }

    // ============================================================================
    // FFMPEG SINGLETON — Pattern 1 (toBlobURL load, pinned core 0.12.10)
    // ============================================================================

    const FFMPEG_CORE_BASE_URL =
      'https://cdn.jsdelivr.net/npm/@ffmpeg/core@0.12.10/dist/umd'

    let ffmpegInstance: FFmpeg | null = null
    let ffmpegLoadPromise: Promise<FFmpeg> | null = null

    export async function getFFmpeg(): Promise<FFmpeg> {
      if (ffmpegInstance) return ffmpegInstance
      if (ffmpegLoadPromise) return ffmpegLoadPromise

      const pre = canRunEngine()
      if (!pre.ok) throw new Error(`ffmpeg unavailable: ${pre.reason}`)

      ffmpegLoadPromise = (async () => {
        const ff = new FFmpeg()
        await ff.load({
          coreURL: await toBlobURL(`${FFMPEG_CORE_BASE_URL}/ffmpeg-core.js`, 'text/javascript'),
          wasmURL: await toBlobURL(`${FFMPEG_CORE_BASE_URL}/ffmpeg-core.wasm`, 'application/wasm'),
        })
        ffmpegInstance = ff
        return ff
      })()

      return ffmpegLoadPromise
    }

    // ============================================================================
    // WARMUP — D-07 background pre-warm (TF.js + COCO-SSD detector hooks added in 03-05)
    // ============================================================================

    /** Pre-warm ffmpeg + (later) TF.js models. Safe to call multiple times. */
    export async function warmup(): Promise<void> {
      // Plans 03-04..03-06 will add TF.js + COCO-SSD warm calls here.
      await getFFmpeg()
    }

    // ============================================================================
    // ANALYSE — D-14 single orchestrator (skeleton; pipeline filled in 03-04..03-06)
    // ============================================================================

    /**
     * Run the full Phase 3 analysis pipeline. The skeleton enforces:
     *   1. Preflight (throws typed error if env unfit)
     *   2. Pre-warm await (D-07 queue behaviour per Open Question 4)
     *   3. Progress callback wiring
     *   4. MEMFS cleanup in finally (Pitfall 2)
     *
     * Pipeline body (probe / scenes / frames / faces / objects / audio / brightness)
     * is filled in by plans 03-04, 03-05, 03-06. Until then, throws a clear error so
     * callers know the work is staged but not implemented.
     */
    export async function analyse(
      file: File,
      opts: AnalyseOptions = {},
    ): Promise<EngineSignals> {
      const pre = canRunEngine()
      if (!pre.ok) throw new Error(`engine unavailable: ${pre.reason}`)

      const onProgress: (s: ProgressStep) => void = opts.onProgress ?? (() => {})

      await warmup()
      const ff = await getFFmpeg()

      const bytes = new Uint8Array(await file.arrayBuffer())
      await ff.writeFile('input.mp4', bytes)

      try {
        // Plans 03-04..03-06 replace this throw with the real pipeline.
        // Keeping it explicit prevents silent "empty signals" regressions.
        onProgress('metadata')
        throw new Error('analyse pipeline not implemented yet — see plans 03-04..03-06')
      } finally {
        await safeDelete(ff, ['input.mp4', 'meta.json'])
        for (let i = 1; i <= 15; i++) {
          await safeDelete(ff, [`frame_${String(i).padStart(3, '0')}.jpg`])
        }
      }
    }

    async function safeDelete(ff: FFmpeg, names: string[]): Promise<void> {
      for (const n of names) {
        try {
          await ff.deleteFile(n)
        } catch {
          // file may not exist — ignore
        }
      }
    }

    // Test-only hook: reset the singleton between tests. Unused in production.
    export function __resetEngineForTests(): void {
      ffmpegInstance = null
      ffmpegLoadPromise = null
    }
    ```
