Motion score formula (RESEARCH §"Pattern 6"):
- For each frame: pick largest-area COCO bbox; centroid = ((x+w/2)/canvasW, (y+h/2)/canvasH)
- For consecutive non-null centroid pairs: distance = hypot(dx, dy)
- motionScore = clamp01(mean(distances) * 4)
- Edge cases: 0 frames or 0 pairs → motionScore = 0

Face confidence aggregation:
- Each face has a `score`/`probability`. faceCount = max across frames; faceConfidence = mean across all detected faces in any frame, default 0 if none.

    ```typescript
    function decodeFrameToCanvas(dataUri: string): Promise<HTMLCanvasElement> {
      return new Promise((resolve, reject) => {
        const img = new Image()
        img.onload = () => {
          const c = document.createElement('canvas')
          c.width = img.naturalWidth
          c.height = img.naturalHeight
          const ctx = c.getContext('2d')
          if (!ctx) {
            reject(new Error('decodeFrameToCanvas: no 2d context'))
            return
          }
          ctx.drawImage(img, 0, 0)
          resolve(c)
        }
        img.onerror = () => reject(new Error('decodeFrameToCanvas: image load failed'))
        img.src = dataUri
      })
    }
    ```

    ```typescript
    // ============================================================================
    // INTERNAL: face detection across frames
    // ============================================================================

    async function detectFacesAcrossFrames(
      frames: HTMLCanvasElement[],
    ): Promise<{ faceCount: number; faceConfidence: number }> {
      if (frames.length === 0) return { faceCount: 0, faceConfidence: 0 }
      const det = await getFaceDetector()

      let maxFaces = 0
      const allConfidences: number[] = []
      for (const c of frames) {
        // Element passing — D-16. Detector manages its own tensors internally.
        const faces = await det.estimateFaces(c)
        if (faces.length > maxFaces) maxFaces = faces.length
        for (const f of faces) {
          // MediaPipe face-detection results expose either `score` or `probability`
          const score = (f as { score?: number }).score
            ?? (f as { probability?: number }).probability
            ?? 0
          if (Number.isFinite(score)) allConfidences.push(score)
        }
      }
      const faceConfidence = allConfidences.length === 0
        ? 0
        : allConfidences.reduce((a, b) => a + b, 0) / allConfidences.length
      return { faceCount: maxFaces, faceConfidence }
    }

    // ============================================================================
    // INTERNAL: object detection + motion score
    // ============================================================================

    async function detectObjectsAndMotion(
      frames: HTMLCanvasElement[],
    ): Promise<{ objectLabels: string[]; motionScore: number }> {
      if (frames.length === 0) return { objectLabels: [], motionScore: 0 }
      const det = await getCocoDetector()
      const labelSet = new Set<string>()
      const centroids: Array<{ x: number; y: number } | null> = []

      for (const c of frames) {
        // detect(canvas, maxBoxes, minScore)
        const preds = await det.detect(c, 5, 0.5)
        preds.forEach((p) => labelSet.add(p.class))
        if (preds.length === 0) {
          centroids.push(null)
          continue
        }
        // Largest-area bbox = primary subject
        const top = preds.slice().sort(
          (a, b) => b.bbox[2] * b.bbox[3] - a.bbox[2] * a.bbox[3],
        )[0]
        const [x, y, w, h] = top.bbox
        centroids.push({
          x: (x + w / 2) / c.width,
          y: (y + h / 2) / c.height,
        })
      }

      let total = 0
      let pairs = 0
      for (let i = 1; i < centroids.length; i++) {
        const a = centroids[i - 1]
        const b = centroids[i]
        if (!a || !b) continue
        total += Math.hypot(a.x - b.x, a.y - b.y)
        pairs++
      }
      const motionScore = pairs === 0 ? 0 : Math.min(1, (total / pairs) * 4)

      return { objectLabels: Array.from(labelSet), motionScore }
    }
    ```
