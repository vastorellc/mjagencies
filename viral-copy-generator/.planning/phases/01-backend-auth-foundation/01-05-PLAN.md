**Step 4: Create Nginx config scaffold for production**

Create `nginx/vcg.conf` with COOP/COEP headers for production Nginx deployment.
This file is not activated in Phase 1 but documents the production config needed before
Phase 3 (ffmpeg requires COOP/COEP in production):

```nginx
# nginx/vcg.conf
# Viral Copy Generator — Nginx configuration
# Deploy to: /etc/nginx/sites-available/vcg.conf
# Symlink: ln -s /etc/nginx/sites-available/vcg.conf /etc/nginx/sites-enabled/vcg.conf

server {
    listen 80;
    server_name your-vps-domain.com;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name your-vps-domain.com;

    # SSL certificates (use certbot: sudo certbot --nginx)
    ssl_certificate     /etc/letsencrypt/live/your-vps-domain.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/your-vps-domain.com/privkey.pem;

    # COOP/COEP — required for SharedArrayBuffer (ffmpeg.wasm) in Phase 3
    # These headers MUST be present on every response including the SPA entry point
    add_header Cross-Origin-Opener-Policy "same-origin" always;
    add_header Cross-Origin-Embedder-Policy "require-corp" always;
    add_header Cross-Origin-Resource-Policy "same-origin" always;

    # Backend API proxy
    location /api/ {
        proxy_pass http://127.0.0.1:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }

    # Health check (no auth)
    location /health {
        proxy_pass http://127.0.0.1:3001;
    }

    # VPS file storage — served directly by Nginx (STORE-02)
    # Meta platforms fetch video from this URL for Instagram/Facebook uploads
    location /uploads/ {
        alias /var/uploads/;
        internal;  # Only accessible via X-Accel-Redirect — not directly browsable
        add_header Cross-Origin-Resource-Policy "same-origin" always;
    }

    # Frontend static files (after Phase 10 build)
    location / {
        root /var/www/vcg;
        try_files $uri $uri/ /index.html;
        add_header Cross-Origin-Opener-Policy "same-origin" always;
        add_header Cross-Origin-Embedder-Policy "require-corp" always;
    }
}
```

Note: The Nginx config uses `internal` on `/uploads/` to prevent direct public access.
For Meta video fetch (Phase 6), the backend returns an X-Accel-Redirect header to serve
the file. This is more secure than `alias /var/uploads/` without `internal`.
