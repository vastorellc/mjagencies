export const COVER_THRESHOLDS = {
  CONTRAST_STDEV_LOW: 0.08,         // luma stdev → score 0 (flat/boring)
  CONTRAST_STDEV_HIGH: 0.25,        // luma stdev → score 100 (punchy)
  FACE_AREA_MIN: 0.08,              // face area / frame area → faceOn=100
  EYE_CONTACT_BAND: 0.15,           // iris centroid must be within ±15% of face width from center
  MOTION_BLUR_LAPLACIAN_LOW: 0.001, // Laplacian variance → score 0 (blurry) [calibrate via D-07]
  MOTION_BLUR_LAPLACIAN_HIGH: 0.02, // Laplacian variance → score 100 (sharp) [calibrate via D-07]
  TEXT_ZONE_VARIANCE_HIGH: 0.05,    // row-luma variance → score 0 (non-uniform) [calibrate via D-07]
} as const

  faceOn: 'Face center',
  ruleOfThirds: 'Rule of thirds',
  contrast: 'High contrast',
  textZone: 'Clean text zone',
  motionBlur: 'Sharp focus',
  eyeContact: 'Eye contact',
}

const LAPLACIAN_KERNEL = [0, 1, 0, 1, -4, 1, 0, 1, 0]

/**
 * scoreContrastFromData — COVER-02c
 * Computes luma standard deviation on pixel data, maps to [0,100] via piecewise linear.
 * stdev 0.08 → 0, stdev 0.25 → 100.
 * Source: RESEARCH.md Pattern 1
 */
export function scoreContrastFromData(data: Uint8ClampedArray, width: number, height: number): number {
  let sum = 0
  let sumSq = 0
  const total = width * height
  if (total === 0) return 0
  for (let i = 0; i < data.length; i += 4) {
    const luma = (0.299 * data[i] + 0.587 * data[i + 1] + 0.114 * data[i + 2]) / 255
    sum += luma
    sumSq += luma * luma
  }
  const mean = sum / total
  const variance = sumSq / total - mean * mean
  const stdev = Math.sqrt(Math.max(0, variance))
  const LOW = COVER_THRESHOLDS.CONTRAST_STDEV_LOW
  const HIGH = COVER_THRESHOLDS.CONTRAST_STDEV_HIGH
  return Math.max(0, Math.min(100, ((stdev - LOW) / (HIGH - LOW)) * 100))
}

/**
 * scoreMotionBlurFromData — COVER-02e
 * Applies 3×3 Laplacian kernel to grayscale luma, computes variance. High variance = sharp.
 * Source: RESEARCH.md Pattern 3 (Revolut Tech blog — Laplacian blur detection)
 */
export function scoreMotionBlurFromData(data: Uint8ClampedArray, width: number, height: number): number {
  const gray: number[] = []
  for (let i = 0; i < data.length; i += 4) {
    gray.push((0.299 * data[i] + 0.587 * data[i + 1] + 0.114 * data[i + 2]) / 255)
  }
  const laplacian: number[] = []
  for (let y = 1; y < height - 1; y++) {
    for (let x = 1; x < width - 1; x++) {
      const idx = y * width + x
      const v =
        LAPLACIAN_KERNEL[0] * gray[idx - width - 1] +
        LAPLACIAN_KERNEL[1] * gray[idx - width] +
        LAPLACIAN_KERNEL[2] * gray[idx - width + 1] +
        LAPLACIAN_KERNEL[3] * gray[idx - 1] +
        LAPLACIAN_KERNEL[4] * gray[idx] +
        LAPLACIAN_KERNEL[5] * gray[idx + 1] +
        LAPLACIAN_KERNEL[6] * gray[idx + width - 1] +
        LAPLACIAN_KERNEL[7] * gray[idx + width] +
        LAPLACIAN_KERNEL[8] * gray[idx + width + 1]
      laplacian.push(v)
    }
  }
  const n = laplacian.length
  if (n === 0) return 0
  const mean = laplacian.reduce((a, b) => a + b, 0) / n
  const variance = laplacian.reduce((s, v) => s + (v - mean) ** 2, 0) / n
  const LOW = COVER_THRESHOLDS.MOTION_BLUR_LAPLACIAN_LOW
  const HIGH = COVER_THRESHOLDS.MOTION_BLUR_LAPLACIAN_HIGH
  return Math.max(0, Math.min(100, ((variance - LOW) / (HIGH - LOW)) * 100))
}

/**
 * scoreTextZoneFromData — COVER-02d
 * Evaluates top-third and bottom-third row-mean luma variance; lower variance = better text zone.
 * Source: RESEARCH.md Pattern 6
 */
export function scoreTextZoneFromData(data: Uint8ClampedArray, width: number, height: number): number {
  const scores = [
    textZoneBandScore(data, width, height, 0, Math.floor(height / 3)),
    textZoneBandScore(data, width, height, Math.floor((2 * height) / 3), height),
  ]
  return Math.max(...scores)
}

function textZoneBandScore(
  data: Uint8ClampedArray,
  width: number,
  height: number,
  startRow: number,
  endRow: number,
): number {
  if (startRow >= endRow || width === 0) return 0
  const rowMeans: number[] = []
  for (let y = startRow; y < endRow; y++) {
    let rowSum = 0
    for (let x = 0; x < width; x++) {
      const i = (y * width + x) * 4
      rowSum += (0.299 * data[i] + 0.587 * data[i + 1] + 0.114 * data[i + 2]) / 255
    }
    rowMeans.push(rowSum / width)
  }
  if (rowMeans.length === 0) return 0
  const mean = rowMeans.reduce((a, b) => a + b, 0) / rowMeans.length
  const variance = rowMeans.reduce((s, v) => s + (v - mean) ** 2, 0) / rowMeans.length
  const HIGH = COVER_THRESHOLDS.TEXT_ZONE_VARIANCE_HIGH
  return Math.max(0, Math.min(100, Math.round(100 * (1 - variance / HIGH))))
}

/**
 * scoreRuleOfThirds — COVER-02b
 * Distance of dominant subject centroid from nearest of 4 power points, normalized.
 * Formula (SPEC locked): 100 − (normalizedDist × 200), clamped to [0,100].
 * Source: RESEARCH.md Pattern 5 + SPEC requirement 2
 */
export function scoreRuleOfThirds(
  centroidX: number,
  centroidY: number,
  frameWidth: number,
  frameHeight: number,
): number {
  if (frameWidth === 0 || frameHeight === 0) return 0
  const powerPoints = [
    { x: frameWidth / 3, y: frameHeight / 3 },
    { x: (2 * frameWidth) / 3, y: frameHeight / 3 },
    { x: frameWidth / 3, y: (2 * frameHeight) / 3 },
    { x: (2 * frameWidth) / 3, y: (2 * frameHeight) / 3 },
  ]
  const diagonal = Math.sqrt(frameWidth ** 2 + frameHeight ** 2)
  let minDist = Infinity
  for (const pp of powerPoints) {
    const d = Math.sqrt((centroidX - pp.x) ** 2 + (centroidY - pp.y) ** 2)
    if (d < minDist) minDist = d
  }
  const normalized = minDist / diagonal
  return Math.max(0, Math.min(100, Math.round(100 - normalized * 200)))
}

/**
 * scoreFaceOn — COVER-02a
 * Face bounding box present + face area ≥ 8% of frame area → 100. Absent or too small → 0.
 * bbox is from @tensorflow-models/face-detection Face.box (pixel-space).
 */
export function scoreFaceOn(bbox: FaceBbox | null, frameArea: number): number {
  if (!bbox || frameArea === 0) return 0
  const faceArea = bbox.width * bbox.height
  if (faceArea / frameArea >= COVER_THRESHOLDS.FACE_AREA_MIN) return 100
  return 0
}

/**
 * scoreEyeContact — COVER-02f
 * MediaPipe FaceMesh iris centroid (indices 468–477) vs face bbox center.
 * Score 100 when within ±15% of face width; falls linearly to 0 at 2× that band.
 * landmarks are NormalizedLandmarkList (x,y in [0,1] × frame dimensions).
 * Source: RESEARCH.md Pattern 4 + SPEC requirement 2
 */
export function scoreEyeContact(
  landmarks: NormalizedLandmarkList | null,
  faceBbox: { xMin: number; xMax: number; width: number } | null,
  frameWidth: number,
): number {
  if (!landmarks || !faceBbox || landmarks.length < 473) return 0
  let irisSumX = 0
  for (let i = 468; i <= 477; i++) irisSumX += landmarks[i].x * frameWidth
  const irisX = irisSumX / 10
  const faceCenterX = (faceBbox.xMin + faceBbox.xMax) / 2
  const band = COVER_THRESHOLDS.EYE_CONTACT_BAND * faceBbox.width
  const dist = Math.abs(irisX - faceCenterX)
  if (dist <= band) return 100
  const falloffEnd = band * 2
  if (falloffEnd <= band) return 0
  return Math.max(0, Math.round(100 * (1 - (dist - band) / (falloffEnd - band))))
}

/**
 * computeMeanLuma — BT.601 formula; same as engine.ts:652
 */
function computeMeanLuma(data: Uint8ClampedArray): number {
  if (data.length === 0) return 0
  let sum = 0
  let count = 0
  for (let i = 0; i < data.length; i += 4) {
    sum += (0.299 * data[i] + 0.587 * data[i + 1] + 0.114 * data[i + 2]) / 255
    count++
  }
  return count > 0 ? sum / count : 0
}

/**
 * pickTextColor — COVER-07
 * luma ≥ 0.55 → black fill + white stroke (dark text on bright background)
 * luma < 0.55  → white fill + black stroke (light text on dark background)
 */
export function pickTextColor(meanLuma: number): [fill: string, stroke: string] {
  return meanLuma >= 0.55 ? ['#000000', '#ffffff'] : ['#ffffff', '#000000']
}

/**
 * truncateText — single-line truncation with ellipsis to fit maxWidth
 */
function truncateText(ctx: CanvasRenderingContext2D, text: string, maxWidth: number): string {
  if (ctx.measureText(text).width <= maxWidth) return text
  let truncated = text
  while (truncated.length > 0 && ctx.measureText(truncated + '…').width > maxWidth) {
    truncated = truncated.slice(0, -1)
  }
  return truncated + '…'
}

export async function getFaceMeshDetector(): Promise<FaceMeshInstance | null> {
  if (faceMeshInstance) return faceMeshInstance
  if (faceMeshPromise) return faceMeshPromise
  faceMeshPromise = (async () => {
    try {
      // Dynamic import so @mediapipe/face_mesh is excluded from initial bundle (lazy-load)

        locateFile: (file: string) =>

      mesh.setOptions({
        maxNumFaces: 1,
        refineLandmarks: true,      // D-06: MANDATORY for iris landmark indices 468–477
        minDetectionConfidence: 0.5,
        minTrackingConfidence: 0.5,
      })
      // FaceMesh uses callback API — must promisify init (Pitfall 1)
      // ORDERING NOTE: timeout is declared BEFORE mesh.send() so clearTimeout ref is always valid
      await new Promise<void>((resolve, reject) => {
        const timeout = setTimeout(
          () => reject(new Error('FaceMesh init timeout')),
          15_000,
        )
        mesh.onResults(() => {
          clearTimeout(timeout)
          resolve()
        })
        const blank = document.createElement('canvas')
        blank.width = 1; blank.height = 1
        mesh.send({ image: blank }).catch(reject)
      })
      faceMeshInstance = mesh as unknown as FaceMeshInstance
      return faceMeshInstance
    } catch (err) {
      // D-05: single console.warn; other 5 predictors continue
      console.warn('[cover:facemesh] init failed — eyeContact predictor disabled:', err)
      return null
    }
  })()
  return faceMeshPromise
}

/**
 * estimateIrisLandmarks — per-frame FaceMesh inference with promisified callback.
 * Sequential calls required — never call send() in parallel (Pitfall 1 cross-talk).
 */
async function estimateIrisLandmarks(
  mesh: FaceMeshInstance,
  canvas: HTMLCanvasElement,
): Promise<NormalizedLandmarkList | null> {
  return new Promise((resolve) => {
    mesh.onResults((results: { multiFaceLandmarks?: NormalizedLandmarkList[] }) => {
      const landmarks = results.multiFaceLandmarks?.[0] ?? null
      resolve(landmarks)
    })
    mesh.send({ image: canvas }).catch(() => resolve(null))
  })
}

function zeroCoverFrameScore(frameIndex: number): CoverFrameScore {
  return {
    frameIndex,
    score: 0,
    breakdown: { faceOn: 0, ruleOfThirds: 0, contrast: 0, textZone: 0, motionBlur: 0, eyeContact: 0 },

function deriveBadges(breakdown: CoverFrameBreakdown): string[] {
  // Top 2 predictor labels sorted by score descending
  const entries = Object.entries(breakdown) as [keyof CoverFrameBreakdown, number][]
  entries.sort((a, b) => b[1] - a[1])
  return entries.slice(0, 2).map(([key]) => PREDICTOR_LABELS[key])
}

/**
 * scoreCoverFrames — D-03
 * Orchestrates all 6 predictors for each of the 10 frames.
 * Called from engine.ts analyse() after detectFacesAcrossFrames, before canvases leave scope.
 *
 * @param canvases — HTMLCanvasElement[] from extractFramesViaVideo (length = 10)
 * @param perFrame  — Face[][] from refactored detectFacesAcrossFrames (COVER-01 / D-02)
 * @param faceMeshDetector — FaceMesh singleton or null (eyeContact = 0 on null — D-05)
 */
export async function scoreCoverFrames(
  canvases: HTMLCanvasElement[],
  perFrame: Face[][],
  faceMeshDetector: FaceMeshInstance | null,
): Promise<CoverFrameScore[]> {
  const scores: CoverFrameScore[] = []

  for (let i = 0; i < canvases.length; i++) {
    const canvas = canvases[i]
    try {
      const frameFaces = perFrame[i] ?? []
      const primaryFace = frameFaces[0] ?? null
      const bbox: FaceBbox | null = primaryFace
        ? (primaryFace.box as FaceBbox)
        : null

      const frameArea = canvas.width * canvas.height

      // Face-based predictors (use bbox from face-detection model, pixel-space)
      const faceOn = scoreFaceOn(bbox, frameArea)

      // Rule-of-thirds: use face centroid if present, else frame center
      let centroidX: number
      let centroidY: number
      if (bbox) {
        centroidX = bbox.xMin + bbox.width / 2
        centroidY = bbox.yMin + bbox.height / 2
      } else {
        centroidX = canvas.width / 2
        centroidY = canvas.height / 2
      }
      const ruleOfThirds = scoreRuleOfThirds(centroidX, centroidY, canvas.width, canvas.height)

      // Pixel-based predictors via canvas adapters
      const contrast = scoreContrast(canvas)
      const textZone = scoreTextZone(canvas)
      const motionBlur = scoreMotionBlur(canvas)

      // Eye contact via FaceMesh iris landmarks (sequential — Pitfall 1)
      let eyeContact = 0
      if (faceMeshDetector && bbox) {
        const landmarks = await estimateIrisLandmarks(faceMeshDetector, canvas)
        if (landmarks) {
          const bboxForEye = { xMin: bbox.xMin, xMax: bbox.xMin + bbox.width, width: bbox.width }
          eyeContact = scoreEyeContact(landmarks, bboxForEye, canvas.width)
        }
      }

      const breakdown: CoverFrameBreakdown = { faceOn, ruleOfThirds, contrast, textZone, motionBlur, eyeContact }

      // COVER-03: equal-weight mean of 6 predictors, rounded to integer
      const score = Math.round((faceOn + ruleOfThirds + contrast + textZone + motionBlur + eyeContact) / 6)

      scores.push({
        frameIndex: i,
        score,
        breakdown,
        badges: deriveBadges(breakdown),
      })
    } catch {
      // COVER-08: graceful degradation — zero score on error, never throw
      scores.push(zeroCoverFrameScore(i))
    }
  }

  return scores
}

/**
 * renderCoverPng
 * Letterbox-scales sourceCanvas into {width, height}, then overlays coverText at bottom-third.
 * Auto-contrast: samples mean luma of bottom-third to pick text color (COVER-07).
 * coverText rendered via canvas fillText/strokeText — NOT innerHTML (security).
 * Source: RESEARCH.md Pattern 7
 */
export function renderCoverPng(
  sourceCanvas: HTMLCanvasElement,
  coverText: string,
  options: { width: number; height: number },
): HTMLCanvasElement {
  const { width, height } = options
  const out = document.createElement('canvas')
  out.width = width; out.height = height
  const ctx = out.getContext('2d')!

  // Black letterbox background
  ctx.fillStyle = '#000000'
  ctx.fillRect(0, 0, width, height)

  // Letterbox-scale: fit source into target preserving aspect ratio (COVER-06)
  const srcW = sourceCanvas.width
  const srcH = sourceCanvas.height
  if (srcW > 0 && srcH > 0) {
    const scaleX = width / srcW
    const scaleY = height / srcH
    const scale = Math.min(scaleX, scaleY)
    const drawW = Math.round(srcW * scale)
    const drawH = Math.round(srcH * scale)
    const offsetX = Math.round((width - drawW) / 2)
    const offsetY = Math.round((height - drawH) / 2)
    ctx.drawImage(sourceCanvas, offsetX, offsetY, drawW, drawH)
  }

  // Auto-contrast text overlay (COVER-07) — only when coverText is non-empty
  if (coverText.trim()) {
    const fontPx = Math.round(height * 0.07)
    // Sample mean luma of the bottom-third
    const bottomY = Math.round((height * 2) / 3)
    const bottomH = Math.round(height / 3)
    const bottomData = ctx.getImageData(0, bottomY, width, bottomH)
    const meanLuma = computeMeanLuma(bottomData.data)
    const [fill, stroke] = pickTextColor(meanLuma)

    ctx.font = `bold ${fontPx}px sans-serif`
    ctx.textAlign = 'center'
    ctx.textBaseline = 'bottom'
    const maxW = width * 0.85
    const truncated = truncateText(ctx, coverText, maxW)

    ctx.strokeStyle = stroke
    ctx.lineWidth = 2
    ctx.strokeText(truncated, width / 2, height - fontPx * 0.5)
    ctx.fillStyle = fill
    ctx.fillText(truncated, width / 2, height - fontPx * 0.5)
  }

  return out
}
