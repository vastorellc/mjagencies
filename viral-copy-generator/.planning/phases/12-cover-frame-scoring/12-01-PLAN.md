// This test runs in browser mode (Playwright/Chromium) because it needs real canvas
// pixel rendering for accurate per-frame scores on real fixture videos.
import { describe, it, expect } from 'vitest'
import { COVER_THRESHOLDS, scoreCoverFrames, getFaceMeshDetector } from './cover'

const FIXTURES = ['with-face.mp4', 'no-face.mp4', 'no-audio.mp4', 'sample.mov'] as const

// Helper: load a fixture File from the test fixtures directory
async function loadFixture(name: string): Promise<File> {
  const url = `/test/fixtures/${name}`
  const resp = await fetch(url)
  if (!resp.ok) throw new Error(`fixture not found: ${name} (${resp.status})`)
  const blob = await resp.blob()
  return new File([blob], name, { type: blob.type || 'video/mp4' })
}

// Helper: extract frames from a video File using HTMLVideoElement + requestVideoFrameCallback
async function extractFramesFromFile(file: File, count = 10): Promise<HTMLCanvasElement[]> {
  return new Promise<HTMLCanvasElement[]>((resolve, reject) => {
    const video = document.createElement('video')
    video.preload = 'auto'
    video.muted = true
    const url = URL.createObjectURL(file)
    const canvases: HTMLCanvasElement[] = []
    let resolved = false

    video.onloadedmetadata = () => {
      const duration = video.duration
      const interval = duration / count
      let frameIdx = 0

      function captureNext(): void {
        if (frameIdx >= count) {
          URL.revokeObjectURL(url)
          resolved = true
          resolve(canvases)
          return
        }
        video.currentTime = frameIdx * interval + interval / 2
      }

      video.onseeked = () => {
        const canvas = document.createElement('canvas')
        canvas.width = video.videoWidth || 512
        canvas.height = video.videoHeight || 512
        const ctx = canvas.getContext('2d')!
        ctx.drawImage(video, 0, 0, canvas.width, canvas.height)
        canvases.push(canvas)
        frameIdx++
        captureNext()
      }

      captureNext()
    }

    video.onerror = () => {
      if (!resolved) {
        URL.revokeObjectURL(url)
        reject(new Error(`Could not load video: ${file.name}`))
      }
    }

    video.src = url
  })
}
