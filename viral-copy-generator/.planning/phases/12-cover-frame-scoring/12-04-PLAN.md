```typescript
/**
 * CoverFramePanel.tsx — Phase 12: Cover-Frame Scoring & Recommendation
 *
 * Displays the top-3 recommended frames as a horizontal carousel with hero + flanking thumbnails.
 * "Show all 10" expands an inline 5×2 grid below. Any frame can be promoted to hero by clicking.
 * cover_text from Phase 5 (aiOutput.instagram.cover_text) is previewed as a canvas overlay on the hero.
 * 4 download buttons produce PNGs at platform-specific dimensions via canvas.toBlob().
 *
 * Security:
 *   - cover_text rendered via canvas fillText in renderCoverPng — NOT innerHTML (immune to XSS)
 *   - URL.revokeObjectURL() called immediately after download trigger (prevents Blob URL leak)
 *   - No cross-origin canvas content (all frames from user's own video — engine.ts:300)
 *
 * D-09: Top-3 carousel with hero (#1 large) + flanking thumbnails (#2, #3)
 * D-10: "Show all 10" inline 5×2 grid, chevron toggle
 * D-11: 4 explicit download buttons, filename pattern: cover-{platform}-{baseName}-{Date.now()}.png
 * D-12: Single renderCoverPng path for both preview AND download (WYSIWYG)
 */
import { useState, useRef, useEffect, useCallback } from 'react'
import type { CoverFrameScore } from '../lib/types'
import { renderCoverPng } from '../lib/cover'

// ============================================================================
// CONSTANTS
// ============================================================================

const DOWNLOAD_SIZES = [
  { label: 'YouTube 1280×720', platform: 'youtube', width: 1280, height: 720 },
  { label: 'Instagram 1080×1080', platform: 'instagram', width: 1080, height: 1080 },
  { label: 'Story / TikTok 1080×1920', platform: 'story', width: 1080, height: 1920 },
  { label: 'Facebook 1280×720', platform: 'facebook', width: 1280, height: 720 },
] as const

// Score band badge color classes — full Tailwind strings for JIT detection (ScorePanel pattern)
const SCORE_BADGE_CLASSES: Record<string, string> = {
  high: 'bg-emerald-700 text-emerald-100',
  mid: 'bg-amber-700 text-amber-100',
  low: 'bg-zinc-700 text-zinc-300',
}

function scoreBadgeClass(score: number): string {
  if (score >= 75) return SCORE_BADGE_CLASSES.high
  if (score >= 50) return SCORE_BADGE_CLASSES.mid
  return SCORE_BADGE_CLASSES.low
}

// ============================================================================
// PROPS
// ============================================================================

interface Props {
  frames: string[]           // framesBase64 — base64 JPEG data URIs from EngineSignals
  scores: CoverFrameScore[]  // coverFrameScores — sorted here by score desc
  coverText: string          // aiOutput?.instagram?.cover_text ?? ''
}

// ============================================================================
// HELPERS
// ============================================================================

/** Convert a base64 data URI to an HTMLCanvasElement (for renderCoverPng) */
function base64ToCanvas(dataUri: string): Promise<HTMLCanvasElement> {
  return new Promise((resolve, reject) => {
    const img = new Image()
    img.onload = () => {
      const canvas = document.createElement('canvas')
      canvas.width = img.naturalWidth || img.width
      canvas.height = img.naturalHeight || img.height
      const ctx = canvas.getContext('2d')!
      ctx.drawImage(img, 0, 0)
      resolve(canvas)
    }
    img.onerror = reject
    img.src = dataUri
  })
}

/** Determine if the scores indicate no faces were detected (all faceOn === 0) */
function hasNoFaces(scores: CoverFrameScore[]): boolean {
  if (scores.length === 0) return false
  return scores.every((s) => s.breakdown.faceOn === 0)
}

// ============================================================================
// COMPONENT
// ============================================================================

export default function CoverFramePanel({ frames, scores, coverText }: Props) {
  // Sort scores descending by score (COVER-04: top-3 by score desc)
  const sorted = [...scores].sort((a, b) => b.score - a.score)
  const top3 = sorted.slice(0, 3)
  const rest = sorted.slice(3)

  // Component-local state (D-09/D-10 discretion area)
  const [heroIdx, setHeroIdx] = useState<number>(0)   // index into sorted[]
  const [expanded, setExpanded] = useState<boolean>(false)

  const noFaces = hasNoFaces(scores)
  const heroScore = sorted[heroIdx]
  const heroFrame = heroScore ? frames[heroScore.frameIndex] : frames[0]

  // Hero canvas preview ref (D-12: canvas not DOM overlay for cover_text)
  const heroCanvasRef = useRef<HTMLCanvasElement>(null)
  const [heroCanvasReady, setHeroCanvasReady] = useState(false)

  // Render the hero frame with cover_text onto the preview canvas (D-12)
  useEffect(() => {
    if (!heroCanvasRef.current || !heroFrame) return
    const canvas = heroCanvasRef.current
    setHeroCanvasReady(false)

    base64ToCanvas(heroFrame)
      .then((sourceCanvas) => {
        // Render at display size (480×270 — 16:9 equivalent at panel width)
        const displayW = canvas.width || 480
        const displayH = canvas.height || 270
        const out = renderCoverPng(sourceCanvas, coverText, { width: displayW, height: displayH })
        const ctx = canvas.getContext('2d')!
        ctx.clearRect(0, 0, canvas.width, canvas.height)
        ctx.drawImage(out, 0, 0, canvas.width, canvas.height)
        setHeroCanvasReady(true)
      })
      .catch(() => {
        // Graceful: show blank canvas on decode failure
        setHeroCanvasReady(true)
      })
  }, [heroFrame, coverText])

  // Download handler (COVER-06, D-11, security: URL.revokeObjectURL after trigger)
  const handleDownload = useCallback(
    async (platform: string, width: number, height: number) => {
      if (!heroFrame) return
      try {
        const sourceCanvas = await base64ToCanvas(heroFrame)
        const out = renderCoverPng(sourceCanvas, coverText, { width, height })
        // Pitfall 6: canvas.toBlob is callback-based — must promisify
        const blob = await new Promise<Blob>((resolve, reject) =>
          out.toBlob(
            (b) => (b ? resolve(b) : reject(new Error('toBlob returned null'))),
            'image/png',
          ),
        )
        const url = URL.createObjectURL(blob)
        const a = document.createElement('a')
        a.href = url
        // D-11 filename pattern: cover-{platform}-{Date.now()}.png
        a.download = `cover-${platform}-${Date.now()}.png`
        a.click()
        // Security: revoke immediately after trigger to prevent Blob URL leak
        URL.revokeObjectURL(url)
      } catch (err) {
        console.warn('[CoverFramePanel] download failed:', err)
      }
    },
    [heroFrame, coverText],
  )

  if (sorted.length === 0) return null

  return (
    <div data-testid="cover-frame-panel" className="flex flex-col gap-4">
      {/* Section header */}
      <div className="flex items-center gap-2 px-1">
        <span className="text-lg">🖼</span>
        <h2 className="text-lg font-semibold text-white">Cover Frame Recommendation</h2>
      </div>

      {/* No-faces note — COVER-08 */}
      {noFaces && (
        <p
          data-testid="no-faces-note"
          className="text-sm text-zinc-400 px-1"
        >
          No faces detected — recommendations based on composition, contrast, sharpness, and text-readable zones.
        </p>
      )}

      {/* Hero + flanking carousel — D-09 */}
      <div className="flex gap-3 items-center overflow-x-auto">
        {/* Hero frame (D-12: canvas preview) */}
        <div
          data-testid="hero-frame"
          className="relative flex-shrink-0 rounded-lg overflow-hidden ring-2 ring-emerald-500"
          style={{ width: 240, height: 135 }}
        >
          <canvas
            ref={heroCanvasRef}
            width={240}
            height={135}
            className="w-full h-full object-cover bg-zinc-900"
          />
          {/* Score overlay */}
          {heroScore && (
            <div className="absolute top-2 left-2 flex items-center gap-1">
              <span
                className={`text-xs font-bold px-1.5 py-0.5 rounded ${scoreBadgeClass(heroScore.score)}`}
                data-testid="hero-score"
              >
                {heroScore.score}
              </span>
              <span className="text-xs text-zinc-300">#1</span>
            </div>
          )}
          {/* Badges */}
          {heroScore && heroScore.badges.length > 0 && (
            <div className="absolute bottom-2 left-2 right-2 flex flex-wrap gap-1">
              {heroScore.badges.map((badge) => (
                <span
                  key={badge}
                  className="text-xs px-1.5 py-0.5 rounded bg-black/60 text-white"
                >
                  {badge}
                </span>
              ))}
            </div>
          )}
          {/* cover_text preview label (D-12: rendered on canvas; DOM indicator for test assertions) */}
          {coverText.trim() && heroCanvasReady && (
            <div data-testid="cover-text-preview" className="sr-only" aria-hidden="true">
              {coverText}
            </div>
          )}
        </div>

        {/* Flanking thumbnails — #2 and #3 from sorted */}
        {top3.slice(1).map((scoreItem, idx) => {
          const sortedIdx = idx + 1
          return (
            <button
              key={scoreItem.frameIndex}
              type="button"
              data-testid={`frame-thumbnail-${sortedIdx}`}
              onClick={() => setHeroIdx(sortedIdx)}
              className="relative flex-shrink-0 rounded-lg overflow-hidden ring-1 ring-zinc-600 hover:ring-zinc-400 transition-all cursor-pointer"
              style={{ width: 140, height: 79 }}
            >
              <img
                src={frames[scoreItem.frameIndex]}
                alt={`Frame ${scoreItem.frameIndex + 1} — score ${scoreItem.score}`}
                className="w-full h-full object-cover bg-zinc-900"
              />
              <div className="absolute top-1 left-1">
                <span
                  className={`text-xs font-bold px-1 py-0.5 rounded ${scoreBadgeClass(scoreItem.score)}`}
                >
                  {scoreItem.score}
                </span>
              </div>
            </button>
          )
        })}
      </div>

      {/* Expand toggle — D-10 */}
      {rest.length > 0 && (
        <button
          type="button"
          data-testid="expand-toggle"
          onClick={() => setExpanded((e) => !e)}
          className="flex items-center gap-2 text-sm text-zinc-400 hover:text-zinc-200 transition-colors w-fit"
          aria-expanded={expanded}
        >
          <span>{expanded ? 'Hide extra frames' : `Show all 10 frames`}</span>
          <span
            className={`inline-block transition-transform duration-150 ${expanded ? 'rotate-180' : 'rotate-0'}`}
            aria-hidden="true"
          >
            ▾
          </span>
        </button>
      )}

      {/* Expanded 5×2 grid — D-10 */}
      {expanded && rest.length > 0 && (
        <div
          data-testid="expanded-grid"
          className="grid grid-cols-5 gap-2"
        >
          {rest.map((scoreItem, idx) => {
            const sortedIdx = idx + 3  // offset from top-3
            return (
              <button
                key={scoreItem.frameIndex}
                type="button"
                data-testid={`frame-thumbnail-${sortedIdx}`}
                onClick={() => {
                  setHeroIdx(sortedIdx)
                  setExpanded(false)  // collapse after promoting
                }}
                className="relative rounded overflow-hidden ring-1 ring-zinc-700 hover:ring-zinc-400 transition-all cursor-pointer aspect-video"
              >
                <img
                  src={frames[scoreItem.frameIndex]}
                  alt={`Frame ${scoreItem.frameIndex + 1} — score ${scoreItem.score}`}
                  className="w-full h-full object-cover bg-zinc-900"
                />
                <div className="absolute top-1 left-1">
                  <span className="text-xs font-bold px-1 py-0.5 rounded bg-black/60 text-white">
                    {scoreItem.score}
                  </span>
                </div>
              </button>
            )
          })}
        </div>
      )}

      {/* Download buttons — D-11, COVER-06 */}
      <div className="flex flex-wrap gap-2">
        {DOWNLOAD_SIZES.map(({ label, platform, width, height }) => (
          <button
            key={platform}
            type="button"
            data-testid={`download-${platform}`}
            onClick={() => void handleDownload(platform, width, height)}
            className="flex items-center gap-1 px-3 py-1.5 text-sm rounded-lg bg-zinc-800 hover:bg-zinc-700 text-zinc-200 transition-colors"
          >
            <span aria-hidden="true">⬇</span>
            <span>{label}</span>
          </button>
        ))}
      </div>
    </div>
  )
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| coverText prop → canvas fillText (via renderCoverPng) | User-supplied AI text enters canvas rendering path |
| canvas.toBlob → URL.createObjectURL → a.click → URL.revokeObjectURL | Blob URL lifecycle: create → use → immediately revoke |
| base64 frames → img.src | User's own video frames decoded from base64 data URIs |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-09 | Spoofing / XSS | coverText → renderCoverPng → canvas fillText | mitigate | renderCoverPng uses canvas fillText/strokeText only — text is rasterized as pixels, never interpreted as HTML. Confirmed via grep: no innerHTML or dangerouslySetInnerHTML in CoverFramePanel.tsx |
| T-12-10 | Tampering / Info Disclosure | Blob URL leak after download | mitigate | URL.revokeObjectURL(url) called in handleDownload immediately after a.click(); revocation is synchronous relative to click event; URL cannot be re-used after this point |
| T-12-11 | Information Disclosure | base64 frame decode via Image element | accept | Frames are from user's own video (crossOrigin='anonymous' set at engine.ts:300); no cross-origin taint; canvas is not shared with any third-party |
</threat_model>
