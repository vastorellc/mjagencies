# Plan 11-03: Meta CAPI Server-Side — Direct Fetch, Hashed user_data, event_id Dedup

## Objective

Server-side Meta Conversions API client. Per D-10, NO browser pixel — all Meta tracking goes through this server-side path. Direct `fetch()` to `https://graph.facebook.com/v22.0/{pixel_id}/events` (NOT facebook-nodejs-business-sdk per RESEARCH §3 — SDK is 3MB and Edge-incompatible). User data (email, phone) is SHA-256-hashed per Meta normalization spec. Each event gets a `crypto.randomUUID()` event_id which doubles as the BullMQ job id for retry idempotency.

Wires two emission points:
- **Lead event** — Phase 9 form submission (after createContact()).
- **Purchase event** — Phase 10 invoice payment success worker (after status flips to paid).

Purpose: REQ-142 (Meta CAPI server-side). Wave-2 plan dependent on Plan 11-07 only (CSP unchanged — no Meta domain since this is purely server-side per D-10).

Output: New `meta-capi.ts` + `meta-capi-queue.ts` modules in `@mjagency/analytics`, modifications to `packages/forms/src/actions/submit-form.ts` and `packages/invoices/src/workers/payment-success.ts`, and BullMQ encrypted queue setup.

```typescript
/**
 * packages/analytics/src/meta-capi.ts
 * REQ-142: Meta Conversions API direct fetch client (NOT facebook-nodejs-business-sdk).
 * D-10: server-side only — NO browser pixel. CSP has no Meta domain.
 *
 * Pitfall 3.2 — phone normalization: E.164 without + prefix, US country code 1
 * Pitfall 3.3 — event_time in Unix seconds, not milliseconds
 * Pitfall 3.4 — test_event_code only set when env var present (dev/staging only)
 * Pitfall 3.5 — at least 1 of (em, ph, ip+ua) required; fall back to ip+ua
 */
import { createHash } from 'node:crypto'
import { getAgencySecret, getAgencySecretOptional } from '@mjagency/config'

export interface CapiUserData {
  em?: string
  ph?: string
  client_ip_address?: string
  client_user_agent?: string
  external_id?: string
}

export interface CapiEvent {
  event_name: 'Lead' | 'Purchase' | 'CompleteRegistration' | 'Subscribe' | 'DeleteUser' | string
  event_id?: string                    // optional — generated if missing
  event_time?: number                  // optional — defaults to now (Unix seconds)
  user_data: CapiUserData              // raw inputs; hashed inside sendCapiEvent
  custom_data?: Record<string, unknown>
}

const META_GRAPH_API_VERSION = 'v22.0'

function sha256Hex(input: string): string {
  return createHash('sha256').update(input).digest('hex')
}

function normalizeEmail(email: string): string {
  return sha256Hex(email.trim().toLowerCase())
}

function normalizePhone(phone: string): string {
  // Pitfall 3.2 — strip non-digits, drop leading US country code 1, prepend '1', then SHA-256.
  // Result: hash of '1XXXXXXXXXX'
  const digits = phone.replace(/\D/g, '').replace(/^1/, '')
  return sha256Hex('1' + digits)
}

export async function sendCapiEvent(agencyId: string, event: CapiEvent): Promise<void> {
  const pixelId = getAgencySecret('META_PIXEL_ID', agencyId)
  const accessToken = getAgencySecret('META_ACCESS_TOKEN', agencyId)
  const testEventCode = getAgencySecretOptional('META_TEST_EVENT_CODE', agencyId)

  // Hash user_data per Meta spec
  const hashed: CapiUserData = {}
  if (event.user_data.em) hashed.em = normalizeEmail(event.user_data.em)
  if (event.user_data.ph) hashed.ph = normalizePhone(event.user_data.ph)
  if (event.user_data.client_ip_address) hashed.client_ip_address = event.user_data.client_ip_address
  if (event.user_data.client_user_agent) hashed.client_user_agent = event.user_data.client_user_agent
  if (event.user_data.external_id) hashed.external_id = event.user_data.external_id

  // Pitfall 3.5: at least one identifier required. Throw if all missing.
  if (!hashed.em && !hashed.ph && !(hashed.client_ip_address && hashed.client_user_agent) && !hashed.external_id) {
    throw new Error('CAPI event requires at least one of: em, ph, ip+ua, external_id')
  }

  const body = {
    data: [{
      event_name: event.event_name,
      event_time: event.event_time ?? Math.floor(Date.now() / 1000), // Pitfall 3.3
      event_id: event.event_id ?? crypto.randomUUID(),
      action_source: 'website' as const,
      user_data: hashed,
      ...(event.custom_data && { custom_data: event.custom_data }),
    }],
    access_token: accessToken,
    ...(testEventCode && { test_event_code: testEventCode }), // Pitfall 3.4
  }

  const url = `https://graph.facebook.com/${META_GRAPH_API_VERSION}/${pixelId}/events`
  const res = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  })

  if (!res.ok) {
    const text = await res.text().catch(() => '')
    throw new Error(`Meta CAPI failed ${res.status}: ${text}`)
  }
}
```

**Step 3 — Create packages/analytics/src/meta-capi-queue.ts.**

```typescript
/**
 * packages/analytics/src/meta-capi-queue.ts
 * REQ-142: BullMQ encrypted queue for Meta CAPI events.
 * sensitiveData: true (PII in user_data even when hashed — project policy).
 * jobId = event_id ensures retries don't double-count via Meta dedup.
 */
import { createEncryptedQueue, createEncryptedWorker } from '@mjagency/queue'
import { REDIS_KEY, createLogger } from '@mjagency/config'
import { sendCapiEvent, type CapiEvent } from './meta-capi.js'

interface CapiJobData {
  agencyId: string
  event: CapiEvent
}

const redisHost = process.env.REDIS_HOST ?? 'localhost'
const redisPort = parseInt(process.env.REDIS_PORT ?? '6379', 10)

export async function enqueueCapiEvent(agencyId: string, event: CapiEvent): Promise<string> {
  const queue = createEncryptedQueue<CapiJobData>('meta-capi', {
    host: redisHost,
    port: redisPort,
    keyPrefix: REDIS_KEY.bullPrefix(agencyId),
  })
  const eventId = event.event_id ?? crypto.randomUUID()
  const eventWithId: CapiEvent = { ...event, event_id: eventId }

  await queue.add(event.event_name, { agencyId, event: eventWithId }, {
    sensitiveData: true, // CLAUDE.md rule 7 — PII even when hashed = sensitive
    jobId: eventId,      // BullMQ once-only processing per id (idempotency)
    attempts: 5,
    backoff: { type: 'exponential', delay: 5000 },
  })
  return eventId
}

export function startCapiWorker(agencyId: string): void {
  const log = createLogger({ service: 'mjagency-meta-capi', agencyId })
  createEncryptedWorker<CapiJobData>(
    'meta-capi',
    async (job) => {
      try {
        await sendCapiEvent(job.data.agencyId, job.data.event)
        log.info({ eventId: job.data.event.event_id, eventName: job.data.event.event_name }, 'CAPI event sent')
      } catch (err) {
        log.error({ err, eventId: job.data.event.event_id }, 'CAPI send failed')
        throw err // BullMQ retries via attempts/backoff
      }
    },
    {
      host: redisHost,
      port: redisPort,
      keyPrefix: REDIS_KEY.bullPrefix(agencyId),
    },
  )
}
```
