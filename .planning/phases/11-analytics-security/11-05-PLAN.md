# Plan 11-05: CCPA / ADA Compliance — Opt-Out + Erasure + Hash-Chain Audit + Receipt PDF

## Objective

The largest plan in Phase 11. Three deliverables:

1. **Consent infrastructure** — `ConsentProvider`/`useConsent()` React Context, `mj_consent` cookie (1-year, SameSite=Lax, Secure, NOT httpOnly), Surface 6 cookie hint banner, all 13 layout integrations.
2. **CCPA opt-out** — Surface 4 footer link with EXACT California Civil Code §1798.135 wording, native `<dialog>` modal, `POST /api/ccpa/opt-out` endpoint in all 13 apps that sets cookie + writes consent_log audit row + enqueues BullMQ fan-out (GA4 User Deletion + Meta CAPI deletion + Clarity Delete API).
3. **CCPA erasure** — Surface 3 erasure form + email-verify confirmation flow (jose, 24h TTL), Surface 5 privacy page with 30-day SLA published, `ccpa_erasure_records` collection (immutable, hash-chain), erasure worker fanning out across 7 systems (Postgres, Redis, R2, GA4, Meta CAPI, Clarity, LiteLLM), legal hold checks (HIPAA, ESIGN Act), pdf-lib receipt PDF, R2 vault storage.

Purpose: REQ-144 (CCPA / ADA compliance tooling — opt-out, deletion, export). Wave-2 plan dependent on Plan 11-07 (CSP allowlist + RUM + csp_reports tables in place). Plans 11-01/02/03 provide the GA4/Clarity/Meta CAPI clients this plan calls for the fan-out.

Output: New `@mjagency/compliance` package, two new Drizzle tables (`ccpa_erasure_records`, `consent_log`) + migration, 13 opt-out API routes, 13 erasure-request API routes, 13 privacy page routes, 13 erasure form routes, 13 erasure confirm routes, layout updates in 13 apps.

```typescript
/**
 * packages/db/src/schema/ccpa-erasure-records.ts
 * REQ-144 D-07: hash-chained audit trail of CCPA erasure operations.
 * Per-agency, RLS, immutable (no UPDATE/DELETE — hash chain integrity).
 */
import { pgTable, uuid, text, timestamp, jsonb, index } from 'drizzle-orm/pg-core'
import { sql } from 'drizzle-orm'

export const ccpaErasureRecords = pgTable(
  'ccpa_erasure_records',
  {
    id: uuid('id').primaryKey().default(sql`gen_random_uuid()`),
    agencyId: uuid('agency_id').notNull(),
    requestId: text('request_id').notNull(),
    system: text('system').notNull(), // postgres | redis | r2 | ga4 | meta_capi | clarity | litellm
    result: jsonb('result').notNull(),
    occurredAt: timestamp('occurred_at', { withTimezone: true }).notNull().defaultNow(),
    prevHash: text('prev_hash'),
    recordHash: text('record_hash').notNull(),
  },
  (t) => [
    index('ccpa_erasure_agency_request_idx').on(t.agencyId, t.requestId),
    index('ccpa_erasure_request_idx').on(t.requestId),
    index('ccpa_erasure_occurred_idx').on(t.occurredAt),
  ],
)

export const ccpaErasureRecordsRlsSql = `
  ALTER TABLE ccpa_erasure_records ENABLE ROW LEVEL SECURITY;
  CREATE POLICY ccpa_erasure_records_agency_iso ON ccpa_erasure_records
    USING (agency_id = (current_setting('app.agency_id', true))::uuid);
  -- Append-only — no DELETE policy, no UPDATE policy.
`
```

```typescript
    prevHash = await writeAuditRow(job.data.agencyId, requestId, 'meta_capi', { ok: true }, prevHash)

    // 6. Clarity
    const clarity = job.data.clarityUserId
      ? await clarityDeleteUser(job.data.agencyId, job.data.clarityUserId)
      : { ok: true, status: 200, errorMessage: 'no clarity user id' }
    prevHash = await writeAuditRow(job.data.agencyId, requestId, 'clarity', clarity, prevHash)

    // 7. LiteLLM
    const llm = await litellmDeleteCalls(job.data.agencyId, email)
    prevHash = await writeAuditRow(job.data.agencyId, requestId, 'litellm', llm, prevHash)

    // 8. Receipt PDF + R2 upload + email
    const pdf = await generateErasureReceiptPdf({
      email, requestId, agencyId: job.data.agencyId, completedAt: new Date(),
      finalRecordHash: prevHash,
    })
    await uploadToR2(`erasure-receipts/${job.data.agencyId}/${requestId}.pdf`, pdf)
    await sendEmail({
      to: email,
      subject: 'CCPA Deletion Receipt',
      bodyText: `Your data deletion request (${requestId}) has been completed. Receipt attached.`,
      attachments: [{ filename: `ccpa-receipt-${requestId}.pdf`, content: Buffer.from(pdf) }],
    })

    log.info({ requestId, finalHash: prevHash }, 'Erasure complete')
  }, {
    host: process.env.REDIS_HOST ?? 'localhost',
    port: parseInt(process.env.REDIS_PORT ?? '6379', 10),
    keyPrefix: REDIS_KEY.bullPrefix(agencyId),
  })
}
```

```typescript
import { randomUUID } from 'node:crypto'
import { createErasureToken } from '@mjagency/compliance'
import { sendEmail } from '@mjagency/email'

export const runtime = 'nodejs'

const AGENCY_SLUG = '{slug}'
const AGENCY_ID = process.env.NEXT_PUBLIC_AGENCY_ID!

export async function POST(req: Request): Promise<Response> {
  const { email } = await req.json()
  if (typeof email !== 'string' || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
    return Response.json({ ok: false, error: 'Invalid email' }, { status: 400 })
  }
  const requestId = randomUUID()
  const token = await createErasureToken(email, AGENCY_ID, requestId)
  const url = `${process.env.PUBLIC_BASE_URL}/privacy/erasure/confirm?token=${encodeURIComponent(token)}`
  await sendEmail({
    to: email,
    subject: 'Confirm your data deletion request',
    bodyText: `Click within 24 hours to confirm: ${url}`,
  })
  return Response.json({ ok: true })
}
```
