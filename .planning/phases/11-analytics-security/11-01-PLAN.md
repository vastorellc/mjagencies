# Plan 11-01: GA4 + GTM Server-Side Container — Per-Agency Properties + Cloudflare Workers Proxy + Data API

## Objective

Stand up the analytics ingestion stack: per-agency GA4 properties, server-side Cloudflare Workers sGTM proxy at `analytics.{subdomain}`, Measurement Protocol fallback for server-fired events (e.g., CCPA opt-out from Plan 11-05), and the GA4 Data API client that powers the Plan 11-04 dashboard's Sessions / Top Pages cards.

Purpose: REQ-140 (GA4 + GTM server-side container). Wave-2 plan dependent on Plan 11-07 (CSP nonce + allowlist) — the GA4 `<Script>` tag must carry the per-request nonce, and `googletagmanager.com` is allowlisted in `script-src` only because Plan 11-07 added it. Plan 11-04 (dashboard) imports `runReport()` from this plan to fill traffic KPI cards.

Output: New `@mjagency/analytics` package, new `getAgencySecret()` helper in `@mjagency/config`, Cloudflare Workers sGTM proxy infra, and `GA4InjectScript` wired into all 13 app layouts (gated on `mj_consent` cookie + nonce).

/**
 * packages/config/src/per-agency-env.ts
 * Per-agency env var lookup helper. Generalizes Phase 7 LiteLLM convention.
 *
 * Slug normalization: agencyId.replaceAll('-','_').toUpperCase()
 * e.g., 'web-ecommerce' → 'WEB_ECOMMERCE'
 * Used by: GA4 secrets (Plan 11-01), Meta CAPI (Plan 11-03), Clarity (Plan 11-02).
 */
export function normalizeSlug(agencyId: string): string {
  return agencyId.replaceAll('-', '_').toUpperCase()
}

export function getAgencySecret(prefix: string, agencyId: string): string {
  const key = `${prefix}_${normalizeSlug(agencyId)}`
  const val = process.env[key]
  if (!val) throw new Error(`Missing env var: ${key}`)
  return val
}

/** Same as getAgencySecret but returns undefined instead of throwing. */
export function getAgencySecretOptional(prefix: string, agencyId: string): string | undefined {
  return process.env[`${prefix}_${normalizeSlug(agencyId)}`]
}

**Step 1 — Create packages/analytics/src/ga4-server.ts.**

```typescript
/**
 * packages/analytics/src/ga4-server.ts
 * REQ-140: server-side GA4 events via Measurement Protocol.
 * Used by Plan 11-05 (CCPA opt-out fires GA4 User Deletion via this path).
 * Pitfall 1.4 mitigated: event_name regex + client_id presence checks before send.
 */
import { getAgencySecret } from '@mjagency/config'

const EVENT_NAME_RE = /^[a-z][a-z0-9_]{0,39}$/

export interface GA4ServerEventInput {
  eventName: string
  clientId: string // anonymous UUID derived from session
  params?: Record<string, string | number | boolean>
}

export async function sendServerEvent(agencyId: string, input: GA4ServerEventInput): Promise<void> {
  if (!EVENT_NAME_RE.test(input.eventName)) {
    throw new Error(`Invalid event_name: ${input.eventName} (must match [a-z_]{1,40})`)
  }
  if (!input.clientId || input.clientId.length < 8) {
    throw new Error('Missing client_id for Measurement Protocol event')
  }

  const measurementId = process.env[`NEXT_PUBLIC_GA4_MEASUREMENT_ID_${agencyId.replaceAll('-','_').toUpperCase()}`]
    ?? process.env.NEXT_PUBLIC_GA4_MEASUREMENT_ID
  const apiSecret = getAgencySecret('GA4_API_SECRET', agencyId)
  if (!measurementId) throw new Error(`Missing NEXT_PUBLIC_GA4_MEASUREMENT_ID for ${agencyId}`)

  const url = `https://www.google-analytics.com/mp/collect?measurement_id=${measurementId}&api_secret=${apiSecret}`
  const body = {
    client_id: input.clientId,
    events: [{ name: input.eventName, params: input.params ?? {} }],
  }
  const res = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  })
  // Pitfall 1.4: Measurement Protocol returns 204 even on bad payloads; log non-2xx
  if (!res.ok) {
    throw new Error(`GA4 Measurement Protocol failed ${res.status}: ${await res.text()}`)
  }
}
```

**Step 2 — Create packages/analytics/src/ga4-data-api.ts.**

```typescript
/**
 * packages/analytics/src/ga4-data-api.ts
 * REQ-140 + REQ-143 dashboard reads.
 * Pitfall 1.3 mitigation: 5-min Redis cache to stay under 25K tokens/day GA4 quota.
 * Used by: Plan 11-04 dashboard sessions card + top pages table.
 */
import { BetaAnalyticsDataClient } from '@google-analytics/data'
import { getAgencySecret } from '@mjagency/config'
import { createHash } from 'node:crypto'
import Redis from 'ioredis'

export interface GA4ReportRequest {
  agencyId: string
  dateRanges: Array<{ startDate: string; endDate: string }>
  dimensions?: Array<{ name: string }>
  metrics: Array<{ name: string }>
  limit?: number
  orderBys?: unknown[]
}

export interface GA4ReportRow {
  dimensionValues: Array<{ value: string }>
  metricValues: Array<{ value: string }>
}

const CACHE_TTL_SEC = 300 // Pitfall 1.3 — 5 minute cache per (agency, request hash)

let _redis: Redis | null = null
function getRedis(): Redis {
  if (!_redis) {
    _redis = new Redis({
      host: process.env.REDIS_HOST ?? 'localhost',
      port: parseInt(process.env.REDIS_PORT ?? '6379', 10),
    })
  }
  return _redis
}

function cacheKey(agencyId: string, request: GA4ReportRequest): string {
  const hash = createHash('sha256').update(JSON.stringify(request)).digest('hex').slice(0, 16)
  return `agency:${agencyId}:dashboard:ga4:${hash}`
}

const _clientByAgency = new Map<string, BetaAnalyticsDataClient>()
function getClient(agencyId: string): BetaAnalyticsDataClient {
  if (!_clientByAgency.has(agencyId)) {
    const keyFile = getAgencySecret('GOOGLE_APPLICATION_CREDENTIALS_PATH', agencyId)
    _clientByAgency.set(agencyId, new BetaAnalyticsDataClient({ keyFilename: keyFile }))
  }
  return _clientByAgency.get(agencyId)!
}

export async function runReport(request: GA4ReportRequest): Promise<GA4ReportRow[]> {
  const redis = getRedis()
  const key = cacheKey(request.agencyId, request)
  const cached = await redis.get(key)
  if (cached) return JSON.parse(cached) as GA4ReportRow[]

  const propertyId = getAgencySecret('GA4_PROPERTY_ID', request.agencyId)
  const client = getClient(request.agencyId)
  const [response] = await client.runReport({
    property: `properties/${propertyId}`,
    dateRanges: request.dateRanges,
    dimensions: request.dimensions,
    metrics: request.metrics,
    limit: request.limit,
    orderBys: request.orderBys as never,
  })

  const rows = (response.rows ?? []).map((r) => ({
    dimensionValues: (r.dimensionValues ?? []).map((d) => ({ value: d.value ?? '' })),
    metricValues: (r.metricValues ?? []).map((m) => ({ value: m.value ?? '' })),
  })) as GA4ReportRow[]

  await redis.setex(key, CACHE_TTL_SEC, JSON.stringify(rows))
  return rows
}
```

**Step 3 — Create packages/analytics/src/ga4-script.tsx (server component).**

```typescript
/**
 * packages/analytics/src/ga4-script.tsx
 * REQ-140 + D-01/D-02 + REQ-146: consent-gated GA4 client script with per-request nonce.
 * Server component — reads cookies + headers (Next.js 15 App Router).
 */
import Script from 'next/script'
import { cookies, headers } from 'next/headers'

export async function GA4InjectScript({ measurementId }: { measurementId: string }): Promise<JSX.Element | null> {
  const consent = (await cookies()).get('mj_consent')?.value
  // D-01/D-02: default-on tracking under CCPA opt-out model
  if (consent === 'tracking_blocked') return null

  const nonce = (await headers()).get('x-nonce') ?? ''

  return (
    <>
      <Script
        src={`https://www.googletagmanager.com/gtag/js?id=${measurementId}`}
        strategy="afterInteractive"
        nonce={nonce}
      />
      <Script id="ga4-init" strategy="afterInteractive" nonce={nonce}>
        {`window.dataLayer = window.dataLayer || [];
function gtag(){dataLayer.push(arguments);}
gtag('js', new Date());
gtag('config', '${measurementId}', { send_page_view: true });`}
      </Script>
    </>
  )
}
```

**Step 2 — Create infra/sgtm-worker/src/index.ts.**

```typescript
/**
 * infra/sgtm-worker/src/index.ts
 * REQ-140: Cloudflare Worker reverse proxy for server-side GTM (sGTM).
 * Forwards analytics.{agency_subdomain}/* requests to per-agency Cloud Run sGTM target.
 * Preserves CF-Connecting-IP via X-Forwarded-For for accurate client geo (Pitfall 1.5).
 */
export interface Env {
  // Per-agency Cloud Run target. Worker reads SGTM_TARGET_<SLUG_UPPER> based on hostname.
  [key: string]: string | undefined
}

export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    const url = new URL(request.url)
    // Hostname pattern: analytics.web-ecommerce.mjagency.com
    const m = url.hostname.match(/^analytics\.(web-[a-z]+)\./)
    if (!m) return new Response('Bad Request: unknown analytics subdomain', { status: 400 })
    const slug = m[1]!
    const slugUpper = slug.replaceAll('-', '_').toUpperCase()
    const target = env[`SGTM_TARGET_${slugUpper}`]
    if (!target) return new Response('sGTM target not configured', { status: 503 })

    const sgtmUrl = `${target}${url.pathname}${url.search}`
    const headers = new Headers(request.headers)
    headers.set('X-Forwarded-For', request.headers.get('CF-Connecting-IP') ?? '')
    headers.set('X-Forwarded-Proto', 'https')

    return fetch(sgtmUrl, {
      method: request.method,
      headers,
      body: request.method === 'GET' || request.method === 'HEAD' ? undefined : request.body,
      redirect: 'manual',
    })
  },
} satisfies ExportedHandler<Env>
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → GA4 collection beacon | Public — anonymous client_id; data goes to googletagmanager.com (D-10 allowlisted) |
| Browser → analytics.{subdomain} (sGTM) | Cloudflare Worker reverse-proxies to Cloud Run sGTM; same-origin to first-party domain |
| Server → google-analytics.com/mp/collect | Server-side Measurement Protocol; api_secret kept in Doppler/Wrangler secret |
| Dashboard → GA4 Data API | Service account JSON server-only; never NEXT_PUBLIC_* |
| CF Worker → Cloud Run sGTM | TLS 1.2+; X-Forwarded-For preserves client IP for geo |
