# Plan 11-04: Analytics Dashboards — Per-Agency + Platform Overview

## Objective

Build Surface 1 (per-agency dashboard at `/admin/dashboard`) and Surface 2 (platform-overview at `/admin/dashboard?view=platform`) as a Payload custom admin view. Hybrid data sources per D-12:

- **Traffic + Top Pages** — GA4 Data API via Plan 11-01 `runReport` (5-min Redis cache).
- **CRM (leads, deals)** — per-agency Postgres aggregates from Phase 9 `contacts`, `deals` tables.
- **Revenue MTD** — per-agency Postgres aggregate from Phase 10 `invoices` table.
- **RUM (LCP/INP/CLS p75)** — per-agency Postgres `web_vitals` table (created by Plan 11-07; data flows from updated WebVitalsReporter that dual-emits via `navigator.sendBeacon('/api/rum', ...)`).

Server-rendered initial paint via Payload custom admin view (Phase 5 admin theming pattern). Client polls `/api/admin/dashboard/metrics` every 60s while `document.visibilityState === 'visible'` (D-14). Per-agency view = session.agencyId scope. Platform view = super_admin only — rolls up across 12 agencies via `Promise.all` (Pitfall 4.3).

Purpose: REQ-143 (per-agency analytics dashboards + platform overview). Wave-3 plan dependent on:
- **Plan 11-07** — web_vitals Postgres table + WebVitalsReporter dual-emit (RUM source).
- **Plan 11-01** — GA4 Data API client (`runReport`) + 5-min Redis cache (Sessions + Top Pages source).
- **Plan 11-05** — ConsentProvider + privacy infrastructure (dashboard does not block on consent; reads aggregated data only — but Plan 11-05 establishes the agencies.legal_hold_rules + ccpa_erasure_records collections that the audit-tab v2 may eventually read).

Output: New `packages/analytics/src/dashboard/` module, new `packages/cms/src/admin-views/` registration, new `packages/ui/src/dashboard/` component library, polling endpoint at `apps/web-main/src/app/api/admin/dashboard/metrics/route.ts`.

**Step 1 — Create packages/analytics/src/dashboard/get-rum-percentiles.ts.**

```typescript
/**
 * packages/analytics/src/dashboard/get-rum-percentiles.ts
 * REQ-143 + Plan 11-07: RUM p75 LCP/INP/CLS over last 24h, per page.
 * Reads from web_vitals table populated by WebVitalsReporter sendBeacon.
 */
import { db } from '@mjagency/db'
import { sql } from 'drizzle-orm'

export type RumMetricName = 'LCP' | 'INP' | 'CLS' | 'FCP' | 'TTFB'

export interface RumPercentile {
  metric: RumMetricName
  pagePath: string | null  // null = aggregated across pages
  p75: number
  rating: 'good' | 'needs-improvement' | 'poor'
}

/** Per-page p75 over last 24h for a metric. Returns top 10 worst pages. */
export async function getRumPerPagePercentiles(
  agencyId: string,
  metric: RumMetricName,
): Promise<RumPercentile[]> {
  // RLS-set agency context
  await db.execute(sql`SELECT set_config('app.agency_id', ${agencyId}, true)`)

  const result = await db.execute<{ page_path: string; p75: number }>(sql`
    SELECT
      page_path,
      percentile_cont(0.75) WITHIN GROUP (ORDER BY value)::double precision AS p75
    FROM web_vitals
    WHERE agency_id = ${agencyId}::uuid
      AND metric_name = ${metric}
      AND created_at > NOW() - INTERVAL '24 hours'
    GROUP BY page_path
    ORDER BY p75 DESC
    LIMIT 10
  `)
  return (result.rows ?? []).map((r) => ({
    metric,
    pagePath: r.page_path,
    p75: r.p75,
    rating: rateMetric(metric, r.p75),
  }))
}

/** Site-wide p75 (aggregated across pages) for a single metric. */
export async function getRumOverallP75(
  agencyId: string,
  metric: RumMetricName,
): Promise<RumPercentile> {
  await db.execute(sql`SELECT set_config('app.agency_id', ${agencyId}, true)`)
  const result = await db.execute<{ p75: number | null }>(sql`
    SELECT percentile_cont(0.75) WITHIN GROUP (ORDER BY value)::double precision AS p75
    FROM web_vitals
    WHERE agency_id = ${agencyId}::uuid
      AND metric_name = ${metric}
      AND created_at > NOW() - INTERVAL '24 hours'
  `)
  const p75 = result.rows[0]?.p75 ?? 0
  return { metric, pagePath: null, p75, rating: rateMetric(metric, p75) }
}

function rateMetric(metric: RumMetricName, value: number): 'good' | 'needs-improvement' | 'poor' {
  // UI-SPEC Surface 1 thresholds
  if (metric === 'LCP') return value <= 2500 ? 'good' : value <= 4000 ? 'needs-improvement' : 'poor'
  if (metric === 'INP') return value <= 200 ? 'good' : value <= 500 ? 'needs-improvement' : 'poor'
  if (metric === 'CLS') return value <= 0.1 ? 'good' : value <= 0.25 ? 'needs-improvement' : 'poor'
  if (metric === 'FCP') return value <= 1800 ? 'good' : value <= 3000 ? 'needs-improvement' : 'poor'
  // TTFB
  return value <= 800 ? 'good' : value <= 1800 ? 'needs-improvement' : 'poor'
}
```

```typescript
/**

 * REQ-143 + D-12: Postgres aggregate queries for CRM + invoicing KPIs.
 * Phase 9 contacts/deals + Phase 10 invoices.

import { db, contacts, deals, invoices } from '@mjagency/db'
import { sql, eq, and, gt, sum, count, inArray, notInArray } from 'drizzle-orm'

export async function getLeadsCount(agencyId: string, days = 7): Promise<number> {
  await db.execute(sql`SELECT set_config('app.agency_id', ${agencyId}, true)`)
  const r = await db.execute<{ c: number }>(sql`
    SELECT count(*)::int AS c FROM contacts
    WHERE agency_id = ${agencyId}::uuid AND created_at > NOW() - (${days}::int || ' days')::interval
  `)
  return r.rows[0]?.c ?? 0
}

export async function getOpenPipelineValue(agencyId: string): Promise<number> {
  await db.execute(sql`SELECT set_config('app.agency_id', ${agencyId}, true)`)
  const r = await db.execute<{ s: number | null }>(sql`
    SELECT COALESCE(sum(value), 0)::double precision AS s FROM deals
    WHERE agency_id = ${agencyId}::uuid AND stage NOT IN ('won','lost')
  `)
  return r.rows[0]?.s ?? 0
}

export async function getOpenPipelineCount(agencyId: string): Promise<number> {
  await db.execute(sql`SELECT set_config('app.agency_id', ${agencyId}, true)`)
  const r = await db.execute<{ c: number }>(sql`
    SELECT count(*)::int AS c FROM deals
    WHERE agency_id = ${agencyId}::uuid AND stage NOT IN ('won','lost')
  `)
  return r.rows[0]?.c ?? 0
}

export async function getRevenueMtd(agencyId: string): Promise<{ amount: number; invoiceCount: number }> {
  await db.execute(sql`SELECT set_config('app.agency_id', ${agencyId}, true)`)
  const r = await db.execute<{ amount: number | null; cnt: number }>(sql`
    SELECT
      COALESCE(sum(amount), 0)::double precision AS amount,
      count(*)::int AS cnt
    FROM invoices
    WHERE agency_id = ${agencyId}::uuid
      AND status = 'paid'
      AND paid_at > date_trunc('month', NOW())
  `)
  return { amount: r.rows[0]?.amount ?? 0, invoiceCount: r.rows[0]?.cnt ?? 0 }
}

/** Week-over-week leads delta. Returns positive number for growth (e.g. +12 means +12% vs prior week). */
export async function getLeadsTrendWoW(agencyId: string): Promise<number> {
  const thisWeek = await getLeadsCount(agencyId, 7)
  await db.execute(sql`SELECT set_config('app.agency_id', ${agencyId}, true)`)
  const r = await db.execute<{ c: number }>(sql`
    SELECT count(*)::int AS c FROM contacts
    WHERE agency_id = ${agencyId}::uuid
      AND created_at BETWEEN NOW() - INTERVAL '14 days' AND NOW() - INTERVAL '7 days'
  `)
  const priorWeek = r.rows[0]?.c ?? 0
  if (priorWeek === 0) return thisWeek > 0 ? 100 : 0
  return Math.round(((thisWeek - priorWeek) / priorWeek) * 100)
}
```

```typescript
/**
 * packages/analytics/src/dashboard/get-ga4-traffic.ts
 * REQ-143 + D-12: GA4 Data API reuse from Plan 11-01 (Sessions + Top Pages with 5-min Redis cache).
 */
import { runReport } from '../ga4-data-api.js'

export interface SessionsData {
  totalSessions: number
  trendWoW: number          // percent change vs prior 7 days
  sparklinePoints: number[] // 7 daily session counts for the 7-day window
}

export async function getSessionsLast7d(agencyId: string): Promise<SessionsData> {
  // Fetch this week + prior week
  const rows = await runReport({
    agencyId,
    dateRanges: [
      { startDate: '7daysAgo', endDate: 'yesterday' },
      { startDate: '14daysAgo', endDate: '8daysAgo' },
    ],
    dimensions: [{ name: 'date' }],
    metrics: [{ name: 'sessions' }],
  })

  // GA4 returns rows from both date ranges interleaved; group by row index
  const thisWeekRows = rows.filter((_r, i) => i < rows.length / 2)
  const priorWeekRows = rows.filter((_r, i) => i >= rows.length / 2)
  const thisTotal = thisWeekRows.reduce((sum, r) => sum + parseInt(r.metricValues[0]?.value ?? '0', 10), 0)
  const priorTotal = priorWeekRows.reduce((sum, r) => sum + parseInt(r.metricValues[0]?.value ?? '0', 10), 0)
  const trend = priorTotal === 0 ? (thisTotal > 0 ? 100 : 0) : Math.round(((thisTotal - priorTotal) / priorTotal) * 100)
  const sparkline = thisWeekRows.map((r) => parseInt(r.metricValues[0]?.value ?? '0', 10))

  return { totalSessions: thisTotal, trendWoW: trend, sparklinePoints: sparkline }
}

export interface TopPage {
  pagePath: string
  sessions: number
  /** LCP p75 from web_vitals join — populated separately by orchestrator */
  lcpP75?: number
}

export async function getTopPages(agencyId: string, limit = 10): Promise<TopPage[]> {
  const rows = await runReport({
    agencyId,
    dateRanges: [{ startDate: '7daysAgo', endDate: 'yesterday' }],
    dimensions: [{ name: 'pagePath' }],
    metrics: [{ name: 'sessions' }],
    limit,
    orderBys: [{ desc: true, metric: { metricName: 'sessions' } }],
  })
  return rows.map((r) => ({
    pagePath: r.dimensionValues[0]?.value ?? '',
    sessions: parseInt(r.metricValues[0]?.value ?? '0', 10),
  }))
}
```

```typescript
import { getSessionsLast7d, getTopPages, type SessionsData, type TopPage } from './get-ga4-traffic.js'
import { getRumOverallP75, type RumPercentile } from './get-rum-percentiles.js'
import {
  getLeadsCount, getLeadsTrendWoW, getOpenPipelineValue, getOpenPipelineCount, getRevenueMtd,
} from './get-postgres-aggregates.js'

export interface PerAgencyMetrics {
  agencyId: string
  sessions: SessionsData | { error: string }
  topPages: TopPage[] | { error: string }
  rum: { lcp: RumPercentile; inp: RumPercentile; cls: RumPercentile }
  leads: { count: number; trendWoW: number }
  pipeline: { value: number; openCount: number }
  revenue: { amountMtd: number; invoiceCount: number }
  generatedAt: string
}

export interface DashboardMetrics {
  view: 'agency' | 'platform'
  agencies: PerAgencyMetrics[]
}

/** Pitfall 4.3: parallel per-agency aggregates via Promise.all. Each query <50ms. */
export async function getDashboardMetrics(agencyIds: string[]): Promise<DashboardMetrics> {
  const agencies = await Promise.all(agencyIds.map(async (agencyId): Promise<PerAgencyMetrics> => {
    // Per-agency: traffic + RUM + CRM + revenue, parallel
    const [sessionsR, topPagesR, lcp, inp, cls, leadsCount, leadsTrend, pipelineValue, openCount, revenue] = await Promise.allSettled([
      getSessionsLast7d(agencyId),
      getTopPages(agencyId, 10),
      getRumOverallP75(agencyId, 'LCP'),
      getRumOverallP75(agencyId, 'INP'),
      getRumOverallP75(agencyId, 'CLS'),
      getLeadsCount(agencyId, 7),
      getLeadsTrendWoW(agencyId),
      getOpenPipelineValue(agencyId),
      getOpenPipelineCount(agencyId),
      getRevenueMtd(agencyId),
    ])

    const sessions = sessionsR.status === 'fulfilled' ? sessionsR.value : { error: String(sessionsR.reason) }
    const topPages = topPagesR.status === 'fulfilled' ? topPagesR.value : { error: String(topPagesR.reason) }

    return {
      agencyId,
      sessions,
      topPages,
      rum: {
        lcp: lcp.status === 'fulfilled' ? lcp.value : { metric: 'LCP', pagePath: null, p75: 0, rating: 'poor' },
        inp: inp.status === 'fulfilled' ? inp.value : { metric: 'INP', pagePath: null, p75: 0, rating: 'poor' },
        cls: cls.status === 'fulfilled' ? cls.value : { metric: 'CLS', pagePath: null, p75: 0, rating: 'poor' },
      },
      leads: {
        count: leadsCount.status === 'fulfilled' ? leadsCount.value : 0,
        trendWoW: leadsTrend.status === 'fulfilled' ? leadsTrend.value : 0,
      },
      pipeline: {
        value: pipelineValue.status === 'fulfilled' ? pipelineValue.value : 0,
```

```typescript
/**
 * packages/ui/src/dashboard/use-dashboard-polling.ts
 * REQ-143 + D-14: 60s client polling, paused while tab not visible.
 * Pitfall 4.4 mitigated: requireSession() runs in /api/admin/dashboard/metrics route.
 */
'use client'

import { useEffect, useState } from 'react'
import type { DashboardMetrics } from '@mjagency/analytics'

export function useDashboardPolling(
  initial: DashboardMetrics,
  viewMode: 'agency' | 'platform',
): DashboardMetrics {
  const [data, setData] = useState<DashboardMetrics>(initial)

  useEffect(() => {
    let active = true
    let inFlight = false

    const poll = async () => {
      if (document.visibilityState !== 'visible') return // D-14
      if (inFlight) return
      inFlight = true
      try {
        const res = await fetch(`/api/admin/dashboard/metrics?view=${viewMode}`, { cache: 'no-store' })
        if (active && res.ok) setData(await res.json() as DashboardMetrics)
      } catch {
        // silent — UI keeps showing last good values + refresh status announces failure
      } finally { inFlight = false }
    }

    const id = setInterval(poll, 60_000)
    const onVis = () => { if (document.visibilityState === 'visible') void poll() }
    document.addEventListener('visibilitychange', onVis)

    return () => {
      active = false
      clearInterval(id)
      document.removeEventListener('visibilitychange', onVis)
    }
  }, [viewMode])

  return data
}
```

```typescript
function Sparkline({ points }: { points: number[] }) {
  if (points.length === 0) return null
  const max = Math.max(...points, 1)
  const segments = points.map((p, i) => `${(i / (points.length - 1)) * 100},${100 - (p / max) * 100}`).join(' ')
  return (
    <svg viewBox="0 0 100 100" preserveAspectRatio="none" className="dashboard-sparkline" role="img">
      <title>{`${points.length}-day trend`}</title>
      <polyline points={segments} />
    </svg>
  )
}
```
