```typescript
/**
 * packages/analytics/src/clarity-init.tsx
 * REQ-141 + D-01/D-02: Microsoft Clarity init gated on mj_consent cookie.
 * Server reads cookie in layout and passes consent flag as prop — no pre-consent flash.
 *
 * Mask Mode = 'Strict' is configured in the Clarity project dashboard (NOT in code).
 * See docs/runbooks/clarity-project-setup.md.
 */
'use client'

import { useEffect } from 'react'
import Clarity from '@microsoft/clarity'

export interface ClarityInitProps {
  projectId: string
  /** Computed server-side from mj_consent cookie. true = D-01 default-on, false = blocked. */
  consent: boolean
  /** Optional: SHA-256 of email for cross-session identification (used by Plan 11-05 erasure flow). */
  customId?: string
}

export function ClarityInit({ projectId, consent, customId }: ClarityInitProps): null {
  useEffect(() => {
    if (!consent) return
    if (!projectId) return
    Clarity.init(projectId)
    Clarity.consent() // signal Clarity's own consent layer (default-on under CCPA opt-out)
    if (customId) {
      Clarity.identify(customId)
    }
  }, [consent, projectId, customId])
  return null
}

/**
 * Helper for emitting custom Clarity events with PII redaction (Phase 7 reuse).
 * Call from client components only (Clarity.event runs in browser).
 */
export async function emitClarityEvent(
  eventName: string,
  data: Record<string, string>,
): Promise<void> {
  // Lazy-import redactPii to avoid bundling the full @mjagency/ai surface in client chunks
  const { redactPii } = await import('@mjagency/ai')
  const redacted: Record<string, string> = {}
  for (const [k, v] of Object.entries(data)) {
    redacted[k] = redactPii(v).redacted
  }
  Clarity.event(eventName, redacted)
}
```

```typescript
/**
 * packages/analytics/src/clarity-delete.ts
 * REQ-141 + REQ-144 (D-03/D-05): Microsoft Clarity Delete API client.
 * Used by Plan 11-05 CCPA opt-out fan-out worker AND CCPA erasure worker.
 *
 * The Delete API requires Clarity's internal user ID (Pitfall 2.1).
 * Plan 11-05 captures this via Clarity.identify(sha256(email)) on session start
 * and stores the (email → clarityUserId) mapping in consent_log.
 */
import { getAgencySecret } from '@mjagency/config'

const CLARITY_DELETE_ENDPOINT = 'https://www.clarity.ms/api/v3/delete'

export interface ClarityDeleteResult {
  ok: boolean
  status: number
  errorMessage?: string
}

export async function clarityDeleteUser(
  agencyId: string,
  clarityUserId: string,
): Promise<ClarityDeleteResult> {
  if (!clarityUserId || clarityUserId.length < 4) {
    return { ok: false, status: 400, errorMessage: 'Missing clarityUserId' }
  }

  const token = getAgencySecret('CLARITY_API_TOKEN', agencyId)
  const res = await fetch(CLARITY_DELETE_ENDPOINT, {
    method: 'POST',
    headers: {
      Authorization: `Bearer ${token}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({ userId: clarityUserId }),
  })

  if (!res.ok) {
    const errorMessage = await res.text().catch(() => '')
    return { ok: false, status: res.status, errorMessage }
  }
  return { ok: true, status: res.status }
}
```
