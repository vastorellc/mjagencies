<objective>
Add four publish-gate validators to enforce anti-fabrication and AI disclosure (REQ-082, REQ-086, REQ-409):

1. **validateStatSources** — scans content for exact stat patterns (e.g. "42%", "$1,250", "73 percent"); each must have a nearby citation link `[text](url)` or `<a href>` within 150 chars. Blocks publish if any unsourced stat found.
2. **validateQuoteSources** — scans for `"..."` quoted text or blockquote nodes (in lexical JSON: `{type:'quote'}`). Each must have an attribution citation within 200 chars. Blocks publish if any unsourced quote found.
3. **validateNoPlaceholders** — scans for case-insensitive patterns: `[insert]`, `[TBD]`, `[TODO]`, `coming soon`, `lorem ipsum`. Blocks publish on any match. (CLAUDE.md §5 CONTENT-COMPLETE rule.)
4. **computeAiContentRatio** — calculates per-field AI content ratio (sums `is_ai_generated` markers / total field count). Sets `ai_content_ratio` (number) on data. If `> 0.70`, also sets `ai_disclosure_required: true`.

Wire all four into pages.ts and posts.ts `beforeOperation` arrays alongside existing Phase 5 validators.

Output: 2 hook files (anti-fab + disclosure), 2 test files (TDD), pages.ts + posts.ts updated to include the new validators and the `ai_content_ratio` + `ai_disclosure_required` fields.
</objective>

       /**
        * packages/cms/src/hooks/anti-fab-validators.ts
        * Phase 7 — anti-fabrication publish gates (REQ-082).
        * Throws on publish, warns on draft (matches Phase 5 pattern).
        */
       import type { CollectionBeforeOperationHook } from 'payload'

       // Stat patterns: bare percentages (excluding ranges), dollar amounts, "N percent", "N out of M"
       const STAT_PATTERNS = [
         /(?<!\\d[-–])\\b(\\d{1,4}(?:\\.\\d+)?)\\s*%(?!\\s*[-–]\\s*\\d)/g,  // 42% but not 30-45%
         /\\$\\s*\\d{1,3}(?:[,.]\\d{3})*(?:\\.\\d+)?(?:\\s*(million|billion|m|bn|k))?/gi,  // $1,250 / $1.5M
         /\\b\\d+(?:\\.\\d+)?\\s+percent\\b/gi,
         /\\b\\d+\\s+out\\s+of\\s+\\d+\\b/gi,
       ]

       const PLACEHOLDER_PATTERNS: RegExp[] = [
         /\\[insert[^\\]]*\\]/i,
         /\\[tbd\\]/i,
         /\\[todo\\]/i,
         /\\bcoming\\s+soon\\b/i,
         /\\blorem\\s+ipsum\\b/i,
       ]

       const CITATION_PATTERN = /\\[[^\\]]+\\]\\([^)]+\\)|<a\\s+href=|"url"\\s*:\\s*"https?:/i
       const STAT_CITATION_WINDOW = 150
       const QUOTE_CITATION_WINDOW = 200

       function flatten(content: unknown): string {
         if (!content) return ''
         return typeof content === 'string' ? content : JSON.stringify(content)
       }

       function hasNearbyCitation(haystack: string, matchIdx: number, matchLen: number, window: number): boolean {
         const start = Math.max(0, matchIdx - window)
         const end = Math.min(haystack.length, matchIdx + matchLen + window)
         const snippet = haystack.slice(start, end)
         return CITATION_PATTERN.test(snippet)
       }

       export const validateStatSources: CollectionBeforeOperationHook = async ({ args, operation }) => {
         if (operation !== 'create' && operation !== 'update') return
         const data = args.data as Record<string, unknown> | undefined
         if (!data) return
         const isPublish = data['status'] === 'published'
         const haystack = flatten(data['content'] ?? data['body'])
         if (!haystack) return

         for (const pattern of STAT_PATTERNS) {
           pattern.lastIndex = 0
           let m: RegExpExecArray | null
           while ((m = pattern.exec(haystack)) !== null) {
             if (!hasNearbyCitation(haystack, m.index, m[0].length, STAT_CITATION_WINDOW)) {
               const msg = `Unsourced statistic detected: "${m[0]}". Every stat needs a citation link within 150 chars (REQ-082).`
               if (isPublish) throw new Error(msg)
               console.warn(`[CMS] ${msg} (draft)`)
               return  // Warn once per draft to avoid noise
             }
           }
         }
       }

       export const validateQuoteSources: CollectionBeforeOperationHook = async ({ args, operation }) => {
         if (operation !== 'create' && operation !== 'update') return
         const data = args.data as Record<string, unknown> | undefined
         if (!data) return
         const isPublish = data['status'] === 'published'
         const haystack = flatten(data['content'] ?? data['body'])
         if (!haystack) return

         // Quoted text: "..." with at least 6 chars between quotes (filters out empty/single-word quoted strings)
         const quotedTextPattern = /"([^"]{6,})"/g
         // Lexical blockquote node
         const blockquotePattern = /"type"\\s*:\\s*"quote"/g

         let m: RegExpExecArray | null
         while ((m = quotedTextPattern.exec(haystack)) !== null) {
           if (!hasNearbyCitation(haystack, m.index, m[0].length, QUOTE_CITATION_WINDOW)) {
             const msg = `Unsourced quote detected: ${m[0].slice(0, 60)}... Quotes need attribution within 200 chars (REQ-082).`
             if (isPublish) throw new Error(msg)
             console.warn(`[CMS] ${msg} (draft)`)
             return
           }
         }

         // Reset lastIndex
         blockquotePattern.lastIndex = 0
         while ((m = blockquotePattern.exec(haystack)) !== null) {
           if (!hasNearbyCitation(haystack, m.index, m[0].length, QUOTE_CITATION_WINDOW)) {
             const msg = `Unsourced blockquote node — needs attribution within 200 chars (REQ-082).`
             if (isPublish) throw new Error(msg)
             console.warn(`[CMS] ${msg} (draft)`)
             return
           }
         }
       }

       export const validateNoPlaceholders: CollectionBeforeOperationHook = async ({ args, operation }) => {
         if (operation !== 'create' && operation !== 'update') return
         const data = args.data as Record<string, unknown> | undefined
         if (!data) return
         const isPublish = data['status'] === 'published'
         const haystack = flatten(data['content'] ?? data['body'])
         if (!haystack) return

         for (const pattern of PLACEHOLDER_PATTERNS) {
           const m = pattern.exec(haystack)
           if (m) {
             const msg = `Placeholder text detected: "${m[0]}". CLAUDE.md §5: NEVER ship placeholders (REQ-082).`
             if (isPublish) throw new Error(msg)
             console.warn(`[CMS] ${msg} (draft)`)
             return
           }
         }
       }

    2. Create `packages/cms/src/hooks/ai-disclosure.ts`:
       ```typescript
       /**
        * packages/cms/src/hooks/ai-disclosure.ts
        * Phase 7 — AI content ratio compute + disclosure flag (REQ-086, REQ-409).
        * Sets data.ai_content_ratio (0..1) and data.ai_disclosure_required (boolean).
        * Threshold: > 0.70 triggers disclosure flag.
        */
       import type { CollectionBeforeOperationHook } from 'payload'

       export const AI_DISCLOSURE_THRESHOLD = 0.70

       const TRACKED_FIELDS = ['title', 'content', 'meta_description', 'aio_tldr'] as const

       export const computeAiContentRatio: CollectionBeforeOperationHook = async ({ args, operation }) => {
         if (operation !== 'create' && operation !== 'update') return
         const data = args.data as Record<string, unknown> | undefined
         if (!data) return

         const aiGeneratedFields = (data['ai_generated_fields'] as string[] | undefined) ?? []

         // Dynamic denominator: count only fields that are present and non-empty in this document
         const populatedTracked = TRACKED_FIELDS.filter(f => {
           const val = (data as Record<string, unknown>)[f]
           return val != null && String(val).trim() !== ''
         }).length
         // Count how many tracked fields are AI-generated
         const aiFields = TRACKED_FIELDS.filter(f => aiGeneratedFields.includes(f))
         const ratio = aiFields.length / Math.max(populatedTracked, 1)

         data['ai_content_ratio'] = Math.min(1, Math.max(0, Number(ratio.toFixed(3))))
         data['ai_disclosure_required'] = ratio > AI_DISCLOSURE_THRESHOLD
       }
       ```
