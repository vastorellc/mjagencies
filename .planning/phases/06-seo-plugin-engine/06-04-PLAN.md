  - geoMentions = sum of all city match counts
  - score = Math.min(100, Math.round((geoMentions / Math.max(config.chunkCountMin, 1)) * 100))

  - config.targetCities = [] → score = 0, findings = [{ rule: 'geo-cities-not-configured', passed: false, detail: 'No target cities configured — add cities to agency settings to enable geo scoring' }]
  - pageType !== 'services' AND config.requiredOnServicePages = true → score = 100, findings = [{ rule: 'geo-not-applicable', passed: true, detail: 'Geo scoring not required for this page type' }]
  - pageType = 'services' AND config.requiredOnServicePages = true AND targetCities configured → run scoring

  Test cases:
  - config.targetCities=[], any content → score=0, findings[0].rule='geo-cities-not-configured'
  - pageType='blog', requiredOnServicePages=true → score=100, findings[0].rule='geo-not-applicable'
  - pageType='services', targetCities=['Austin'], content has 'Austin' in 2 paragraphs, chunkCountMin=2 → score=100

  - pageType='services', targetCities=['Austin'], content has 0 Austin mentions → score=0
  - Findings array includes one entry per city in targetCities with mentionCount
  - City match is case-insensitive: 'austin' in content matches targetCity 'Austin'

  ```typescript
  /**
   * packages/seo/src/plugins/geo-chunking.ts
   * geo-chunking scoring plugin — geographic content density scoring.
   * REQ-070: one of 3 required SEO plugins.
   * Algorithm from RESEARCH.md Pattern 4 [CITED: searchengineland.com/geo-local-seo-future-discovery].
   * Note: ASSUMED scoring thresholds (A2 in RESEARCH.md) — admin-tunable via config.chunkCountMin.
   */
  import type { LexicalExtracts } from '../lexical-parser.js'
  import type { PluginDefaults, PluginResult } from '../engine.js'
  import { registerPlugin } from '../engine.js'

  export interface GeoChunkingConfig {
    targetRadius: number
    targetCities: string[]
    chunkCountMin: number
    requiredOnServicePages: boolean
  }

  export interface GeoChunkingResult extends PluginResult {
    score: number
    geoMentionCount: number
    findings: Array<{ rule: string; passed: boolean; detail: string }>
  }

  export function scoreGeoChunking(
    extracts: LexicalExtracts,
    pageType: string,
    config: GeoChunkingConfig,
  ): GeoChunkingResult {
    // Not applicable check
    if (config.requiredOnServicePages && pageType !== 'services') {
      return {
        score: 100,
        geoMentionCount: 0,
        findings: [{ rule: 'geo-not-applicable', passed: true, detail: 'Geo scoring not required for this page type' }],
      }
    }

    // No cities configured
    if (!config.targetCities || config.targetCities.length === 0) {
      return {
        score: 0,
        geoMentionCount: 0,
        findings: [{ rule: 'geo-cities-not-configured', passed: false, detail: 'No target cities configured — add cities to agency settings to enable geo scoring' }],
      }
    }

    const fullText = [
      ...extracts.headings.map(h => h.text),
      ...extracts.paragraphs,
    ].join(' ').toLowerCase()

    let totalMentions = 0
    const findings: GeoChunkingResult['findings'] = []

    for (const city of config.targetCities) {
      const pattern = new RegExp(`\\b${city.toLowerCase().replace(/[.*+?^${}()|[\]\\]/g, '\\$&')}\\b`, 'g')
      const matches = fullText.match(pattern) ?? []
      const mentionCount = matches.length
      totalMentions += mentionCount
      findings.push({
        rule: 'city-mentions',
        passed: mentionCount > 0,
        detail: mentionCount > 0
          ? `"${city}" mentioned ${mentionCount} time${mentionCount === 1 ? '' : 's'}`
          : `"${city}" not found in content — add city references for local SEO`,
      })
    }

    const score = Math.min(100, Math.round((totalMentions / Math.max(config.chunkCountMin, 1)) * 100))
    return { score, geoMentionCount: totalMentions, findings }
  }

  // Register with engine
  registerPlugin('geo-chunking', (extracts, input, config) =>
    scoreGeoChunking(extracts, input.pageType ?? '', config.geo_chunking)
  )
  ```
