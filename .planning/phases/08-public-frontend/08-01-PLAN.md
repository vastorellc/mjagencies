### Task 1: Create WebVitalsReporter client component in packages/ui

**File:** `packages/ui/src/rum/web-vitals.tsx`

Create a `'use client'` component that dynamically imports `web-vitals` (to avoid SSR cost) and registers all five metrics. The component renders `null` — it is a side-effect-only reporter.

```typescript
// packages/ui/src/rum/web-vitals.tsx
'use client'
import { useEffect } from 'react'
import type { Metric } from 'web-vitals'

export function WebVitalsReporter({
  ga4MeasurementId,
}: {
  ga4MeasurementId: string
}): null {
  useEffect(() => {
    void import('web-vitals').then(({ onLCP, onINP, onCLS, onFCP, onTTFB }) => {
      const send = (metric: Metric): void => {
        window.gtag?.('event', metric.name, {
          value: Math.round(metric.name === 'CLS' ? metric.value * 1000 : metric.value),
          event_category: 'Web Vitals',
          event_label: metric.id,
          non_interaction: true,
          send_to: ga4MeasurementId,
        })
      }
      onLCP(send)
      onINP(send)
      onCLS(send)
      onFCP(send)
      onTTFB(send)
    })
  }, [ga4MeasurementId])
  return null
}
```

Also add `declare global { interface Window { gtag?: (...args: unknown[]) => void } }` in the same file to satisfy TypeScript strict mode.

Export `WebVitalsReporter` from `packages/ui/src/index.ts` (append to the existing barrel):


Add `web-vitals` to `packages/ui/package.json` dependencies if not already present:

### Task 2: Update shared auth middleware to exclude public P0 paths

**File:** `packages/auth/src/middleware.ts`

Read the existing `createAuthMiddleware()` and `config` export in this file. The current matcher is:
```
'/((?!_next/static|_next/image|favicon\\.ico|robots\\.txt|sitemap\\.xml|manifest\\.json|api/|\\(payload\\)/admin).*)'
```

Update the matcher to also exclude public frontend paths. These paths must NOT redirect to `/login`:

- `/404`
- `/_not-found`

```typescript
  matcher: [
    '/((?!_next/static|_next/image|favicon\\.ico|robots\\.txt|sitemap\\.xml|manifest\\.json|api/|\\(payload\\)/admin|$|about$|services(?:/.*)?$|blog(?:/.*)?$|contact$|faq$|privacy$|terms$|404$|_not-found$).*)',
  ],
}
```

Alternatively, add a `PUBLIC_PATHS` set check at the top of the `createAuthMiddleware` function body (before the token check) that does `return NextResponse.next()` for public paths. This approach is cleaner than the regex:

```typescript
const PUBLIC_PATHS: ReadonlySet<string> = new Set(['/', '/about', '/contact', '/faq', '/privacy', '/terms'])

function isPublicPath(pathname: string): boolean {
  if (PUBLIC_PATHS.has(pathname)) return true
  if (pathname.startsWith('/blog')) return true
  if (pathname.startsWith('/services')) return true
  return false
}

export function createAuthMiddleware(): (req: NextRequest) => Promise<NextResponse> {
  return async (req: NextRequest): Promise<NextResponse> => {
    if (isPublicPath(req.nextUrl.pathname)) return NextResponse.next()
    // existing token check with jose...
  }
}
```

Use whichever approach is consistent with the existing implementation. Do NOT break authenticated routes like `/dashboard`, `/admin`, `/settings`.
