    Implementation:
    - Parse --base-domain (required), --timeout (default 30 seconds), --help
    - AGENCY_SLUGS array: ['web-ai','web-branding','web-construction','web-dental','web-ecommerce','web-financial','web-fitness','web-homeservices','web-legal','web-realestate','web-restaurant','web-spa']
    - Each health URL: `https://${slug}.${baseDomain}/api/health`
    - Poll loop:
      ```javascript
      const deadline = Date.now() + timeoutMs
      const failed = new Set(AGENCY_SLUGS)
      while (Date.now() < deadline && failed.size > 0) {
        await new Promise(r => setTimeout(r, 2000))
        for (const slug of [...failed]) {
          try {
            const res = await fetch(`https://${slug}.${baseDomain}/api/health`)
            if (res.status === 200) failed.delete(slug)
          } catch { /* ignore — still unhealthy */ }
        }
        console.log(`[health-check] ${AGENCY_SLUGS.length - failed.size}/${AGENCY_SLUGS.length} healthy`)
      }
      if (failed.size > 0) {
        console.error(`[health-check] FAILED: ${[...failed].join(', ')} did not return 200 within ${timeoutMs/1000}s`)
        console.error(`::error::canary health check failed: ${[...failed].join(', ')}`)
        process.exit(1)
      }
      console.log('[health-check] All 12 agencies healthy.')
      process.exit(0)
      ```

    Create `.github/workflows/canary-deploy.yml`:

    ```yaml
    name: Canary Deploy
    on:
      workflow_dispatch:
        inputs:
          canary_script_name:
            description: 'Cloudflare Worker script name of the new canary build'
            required: true
          stable_script_name:
            description: 'Cloudflare Worker script name of the stable build to roll back to'
            required: true
          base_domain:
            description: 'Base domain for health checks (e.g., mjagency.com)'
            required: true
            default: 'mjagency.com'
          health_check_timeout:
            description: 'Health check timeout in seconds'
            required: false
            default: '30'

    concurrency:
      group: canary-${{ github.ref }}
      cancel-in-progress: false

    env:
      TF_WORKING_DIR: infra/cloudflare/canary
      CANARY_WEIGHT: '5'
      STABLE_WEIGHT: '0'

    jobs:
      build:
        name: Build
        runs-on: ubuntu-latest
        outputs:
          build_tag: ${{ steps.tag.outputs.tag }}
        steps:
          - uses: actions/checkout@v4
          - uses: pnpm/action-setup@v4
            with:
              version: 10
          - uses: actions/setup-node@v4
            with:
              node-version-file: ".nvmrc"
              cache: "pnpm"
          - uses: dopplerhq/cli-action@v3
          - run: pnpm install --frozen-lockfile
          - run: pnpm turbo run build
            env:
              DOPPLER_TOKEN: ${{ secrets.DOPPLER_CI_TOKEN }}
          - id: tag
            run: echo "tag=$(git rev-parse --short HEAD)" >> $GITHUB_OUTPUT

      deploy-canary:
        name: Deploy 5% Canary
        needs: build
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: hashicorp/setup-terraform@v3
            with:
              terraform_version: "1.8.0"
          - name: Terraform Init
            working-directory: ${{ env.TF_WORKING_DIR }}
            run: terraform init
            env:
              CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          - name: Deploy canary at 5% weight
            working-directory: ${{ env.TF_WORKING_DIR }}
            run: |
              terraform apply -auto-approve \
                -var="canary_weight=${{ env.CANARY_WEIGHT }}" \
                -var="canary_script_name=${{ inputs.canary_script_name }}" \
                -var="stable_script_name=${{ inputs.stable_script_name }}"
            env:
              CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          - name: Confirm canary active
            run: echo "Canary deployed at ${{ env.CANARY_WEIGHT }}% for ${{ inputs.canary_script_name }}"

      health-check:
        name: Health Check (30s)
        needs: deploy-canary
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4
            with:
              node-version-file: ".nvmrc"
          - name: Poll /api/health for all 12 agencies
            run: node scripts/canary-health-check.mjs --base-domain=${{ inputs.base_domain }} --timeout=${{ inputs.health_check_timeout }}

      promote:
        name: Promote to 100%
        needs: health-check
        if: success()
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: hashicorp/setup-terraform@v3
            with:
              terraform_version: "1.8.0"
          - name: Terraform Init
            working-directory: ${{ env.TF_WORKING_DIR }}
            run: terraform init
            env:
              CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          - name: Promote canary to 100%
            working-directory: ${{ env.TF_WORKING_DIR }}
            run: |
              terraform apply -auto-approve \
                -var="canary_weight=100" \
                -var="canary_script_name=${{ inputs.canary_script_name }}" \
                -var="stable_script_name=${{ inputs.stable_script_name }}"
            env:
              CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          - run: echo "Canary promoted to 100%. Deployment complete."

      rollback:
        name: Auto-Rollback on Failure
        needs: health-check
        if: failure()
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: hashicorp/setup-terraform@v3
            with:
              terraform_version: "1.8.0"
          - name: Terraform Init
            working-directory: ${{ env.TF_WORKING_DIR }}
            run: terraform init
            env:
              CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          - name: Rollback — set canary weight to 0
            working-directory: ${{ env.TF_WORKING_DIR }}
            run: |
              terraform apply -auto-approve \
                -var="canary_weight=${{ env.STABLE_WEIGHT }}" \
                -var="canary_script_name=${{ inputs.canary_script_name }}" \
                -var="stable_script_name=${{ inputs.stable_script_name }}"
            env:
              CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          - name: Rollback complete — alert
            run: |
              echo "::error::Canary rollback executed. Health check failed for ${{ inputs.canary_script_name }}. Stable build ${{ inputs.stable_script_name }} restored."
              exit 1
    ```
