       import { AGENCIES, AGENCY_PORT_BASE } from '@mjagency/config'

       const PGBOUNCER_PORT_BASE = 6432

       export interface AgencyConnection {
         agencySlug: typeof AGENCIES[number]
         /** PgBouncer port for this agency (transaction mode pool). */
         pgbouncerPort: number
         /** Database name on the underlying Postgres instance. */
         dbName: string
         /** Role name owning this DB. */
         role: string
       }

       export function agencyConnection(slug: typeof AGENCIES[number]): AgencyConnection {
         const idx = AGENCIES.indexOf(slug)
         if (idx === -1) throw new Error(`Unknown agency slug: ${slug}`)
         return {
           agencySlug: slug,
           pgbouncerPort: PGBOUNCER_PORT_BASE + idx,
           dbName: `${slug}_db`,
           role: `${slug}_user`,
         }
       }

       /** Builds DATABASE_URL targeting PgBouncer (transaction-mode pool). */
       export function buildDatabaseUrl(slug: typeof AGENCIES[number], password: string): string {
         const c = agencyConnection(slug)
         return `postgresql://${c.role}:${encodeURIComponent(password)}@127.0.0.1:${c.pgbouncerPort}/${c.dbName}`
       }
