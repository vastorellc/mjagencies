        *   - argv: --update-baseline <path-to-size-current.json>  (used by main.yml on merge)

       const args = process.argv.slice(2)

       if (args[0] === '--update-baseline') {
         const currentPath = args[1]
         if (!currentPath) { console.error('Usage: --update-baseline <path>'); process.exit(1) }
         const current = JSON.parse(readFileSync(currentPath, 'utf8'))
         writeFileSync(baselinePath, JSON.stringify(reshape(current), null, 2))
         console.log(`Updated ${baselinePath}`)
         process.exit(0)

       const baseline = existsSync(baselinePath) ? JSON.parse(readFileSync(baselinePath, 'utf8')) : {}

       let warnings = 0, hardFails = 0
       for (const c of current) {
         const base = baseline[c.name]
         if (typeof base !== 'number') {
           console.log(`::notice::No baseline for "${c.name}" — recorded ${c.size} as initial`)
           continue
         }
         const growth = (c.size - base) / base
         if (growth >= 0.25) {
           console.log(`::error::Bundle "${c.name}" grew ${(growth * 100).toFixed(1)}% (+25% threshold). Baseline=${base}, current=${c.size}.`)
           hardFails++
         } else if (growth >= 0.10) {
           console.log(`::warning::Bundle "${c.name}" grew ${(growth * 100).toFixed(1)}% (+10% threshold). Baseline=${base}, current=${c.size}.`)
           warnings++
         }
       }

       if (hardFails > 0) {
         console.error(`D-16: ${hardFails} hard-fail bundle(s); see annotations`)
         process.exit(1)
       }
       console.log(`D-16: bundle-size OK (${warnings} warning(s))`)

       function reshape(arr: Array<{ name: string; size: number }>): Record<string, number> {
         return Object.fromEntries(arr.map((e) => [e.name, e.size]))
