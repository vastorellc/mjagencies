Ship per-app Cloudflare-compatible Next.js middleware for ALL 12 apps (web-main + 11 agency apps). Middleware extracts the agency slug from the Host header, verifies the access token using `jose` (Edge-compatible), validates the embedded `agencyId` claim matches the subdomain (anti-cross-tenant), injects request-context headers (`x-agency-id`, `x-user-id`, `x-user-role`), and applies security headers on every response. The matcher EXACTLY excludes Payload admin routes (`/(payload)/admin/*`), API routes (`/api/*`), Next.js internals (`/_next/*`), and well-known static files — these need Node runtime and would break if pushed through Edge middleware (CLAUDE.md §4, REQ-030).

Purpose: REQ-029 (Next.js >= 15.2.3 CVE-2025-29927 patch) — middleware bypass is mitigated at three layers: (1) Cloudflare WAF strips `x-middleware-subrequest` at the edge (documented in runbook), (2) Next.js >= 15.2.3 patch (CI gate added by Plan 03-06), (3) `requireSession()` in server actions (Plan 03-05) means middleware is never the sole auth check. REQ-030 (matcher exclusions). REQ-310 (explicit `iss`/`aud` on every verify). REQ-408 (per-agency callback route ships here so subdomain rename redirects work via Cloudflare 301).

Plan-time decisions captured (RESEARCH §6, Pitfall 1):
- **Edge runtime safety:** Middleware imports ONLY from `jose`, `next/server`, and `@mjagency/config` (constants only — never `createLogger` from `packages/config/logger.ts`). Pitfall 1 mitigation. Pino is Node-only and would crash Edge at startup.
- **Single source of truth:** `packages/auth/src/middleware.ts` exports a `createAuthMiddleware()` factory; each app's `middleware.ts` is a 4-line re-export. This way the matcher and verify logic are shared across 12 apps without copy-paste drift.
- **CSP:** Phase 3 ships a static CSP via `applySecurityHeaders`. Phase 11 (M011) ships the nonce-per-request CSP as a Cloudflare Worker (SEC-N3). The static CSP is functional and does not break Pino + OTel + Lexical.

- **REQ-408 (subdomain rename redirects):** A Cloudflare Page Rule (NOT Next.js code) handles `oldslug.brand.com/*` → `newslug.brand.com/*` 301s. Documented in the runbook. The middleware's agency-from-host helper handles unknown slugs with a 404 (NOT a 301) — that's deliberate, since unknown != renamed.

```ts
// packages/auth/src/agency-from-host.ts — Edge-safe (no Node imports)
import { AGENCIES, type AgencySlug } from '@mjagency/config'

const KNOWN: ReadonlySet<string> = new Set<string>(AGENCIES)

export function extractAgencyFromHost(hostHeader: string | null): AgencySlug | null {
  if (!hostHeader) return null
  // Strip port (e.g. ecommerce.localhost:3001 → ecommerce.localhost)
  const host = hostHeader.split(':')[0] ?? ''
  // First label is the subdomain
  const parts     = host.split('.')
  const subdomain = parts.length >= 2 ? parts[0] ?? '' : ''
  return KNOWN.has(subdomain) ? (subdomain as AgencySlug) : null
}
```

Locked middleware factory (RESEARCH §6.2 verbatim):
```ts
// packages/auth/src/middleware.ts
// IMPORTANT: This file runs on Next.js Edge runtime. NO Node-only imports allowed.
// - jose: ✅ Edge-compatible
// - next/server: ✅ Edge-compatible
// - @mjagency/config (constants only): ✅ Edge-compatible
// - @mjagency/config/logger (Pino): ❌ NEVER — Pitfall 1 (Pino is Node-only)
// - @mjagency/db: ❌ NEVER — postgres-js is Node-only
// - bcrypt, ioredis, etc.: ❌ NEVER

import { type NextRequest, NextResponse } from 'next/server'
import { jwtVerify } from 'jose'
import { extractAgencyFromHost } from './agency-from-host.js'
import { applySecurityHeaders } from './security-headers.js'

const JWT_ACCESS_SECRET = new TextEncoder().encode(process.env.JWT_ACCESS_SECRET)
const ACCESS_COOKIE  = process.env.NODE_ENV === 'production' ? '__Host-access' : 'mj-access'

interface AccessClaims {
  sub:      string
  agencyId: string
  role:     string
  jti:      string
  familyId: string
}

export function createAuthMiddleware() {
  return async function middleware(req: NextRequest): Promise<NextResponse> {
    const url       = req.nextUrl.clone()
    const agency    = extractAgencyFromHost(req.headers.get('host'))

    // Unknown subdomain → 404 (subdomain takeover probing defense)
    if (!agency) {
      return new NextResponse(null, { status: 404 })
    }

    // Bypass auth for the login route itself + /sso entry + /auth/callback so the SSO flow can complete
    const path = url.pathname
    if (path === '/login' || path === '/sso' || path.startsWith('/auth/callback')) {
      const passthrough = NextResponse.next()
      passthrough.headers.set('x-agency-id', agency)
      applySecurityHeaders(passthrough)
      return passthrough
    }

    // Read the access cookie (also accept legacy/dev names)
    const token =
      req.cookies.get(ACCESS_COOKIE)?.value
      ?? req.cookies.get('__Host-access')?.value
      ?? req.cookies.get('mj-access')?.value

    if (!token) {
      url.pathname = '/login'
      url.searchParams.set('returnTo', req.nextUrl.pathname + req.nextUrl.search)
      return NextResponse.redirect(url)
    }

    try {
      const { payload } = await jwtVerify<AccessClaims>(token, JWT_ACCESS_SECRET, {
        algorithms: ['HS256'],
        issuer:     'mjagency',
        audience:   'mjagency-api',
      })

      // Anti-cross-tenant: token's agencyId MUST match the subdomain
      if (payload.agencyId !== agency) {
        url.pathname = '/login'
        return NextResponse.redirect(url)
      }

      // Inject request-context headers for server components
      const response = NextResponse.next({
        request: {
          headers: new Headers(req.headers),
        },
      })
      response.headers.set('x-agency-id', agency)
      response.headers.set('x-user-id',   payload.sub  ?? '')
      response.headers.set('x-user-role', payload.role ?? '')

      applySecurityHeaders(response)
      return response
    } catch {
      // Verify failed (expired/invalid). Redirect to login.
      url.pathname = '/login'
      url.searchParams.set('returnTo', req.nextUrl.pathname + req.nextUrl.search)
      return NextResponse.redirect(url)
    }
  }
}

export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon\\.ico|robots\\.txt|sitemap\\.xml|manifest\\.json|api/|\\(payload\\)/admin).*)',
  ],
}
```

```ts
// apps/web-main/src/app/auth/callback/route.ts (and identical for each agency app — Plan 03-04 ships ONE in web-main; agency-app copies are deferred to Plan 03-05 or 03-06 per file ownership rules)
// NOTE FOR PLAN 03-04: This route ONLY ships in apps/web-main as a reference implementation.
// Each agency app gets its own copy in Plan 03-05 (which already touches per-app server actions).
// Documented in 03-04 SUMMARY as a partial-coverage decision.
import 'server-only'
import { NextResponse, type NextRequest } from 'next/server'
import { verifySsoState, setAuthCookies } from '@mjagency/auth'

export async function GET(req: NextRequest): Promise<NextResponse> {
  const url   = new URL(req.url)
  const code  = url.searchParams.get('code')
  const state = url.searchParams.get('state')
  if (!code || !state) {
    return NextResponse.redirect(new URL('/login', req.url))
  }

  // Extract expected agency from host (e.g. ecommerce.brand.com → 'ecommerce')
  const host          = req.headers.get('host') ?? ''
  const expectedSlug  = host.split(':')[0]?.split('.')[0] ?? ''
  const stateResult   = verifySsoState(state, expectedSlug)
  if (!stateResult.valid) {
    return NextResponse.redirect(new URL('/login', req.url))
  }

  // Server-to-server exchange against accounts.brand.com
  const exchangeUrl = `https://${process.env.ACCOUNTS_HOST ?? 'accounts.brand.com'}/api/sso/exchange`
  const res = await fetch(exchangeUrl, {
    method:  'POST',
    headers: {
      'content-type':            'application/json',
      'x-mjagency-internal':     process.env.SSO_INTERNAL_TOKEN ?? '',
    },
    body: JSON.stringify({ code }),
  })
  if (!res.ok) {
    return NextResponse.redirect(new URL('/login', req.url))
  }

  const tokens = await res.json() as { accessToken: string; refreshToken: string }
  await setAuthCookies(tokens.accessToken, tokens.refreshToken)

  // Redirect to validated returnTo
  return NextResponse.redirect(new URL(stateResult.returnTo, req.url))
}
```
